# WordMagicGame V0.3.5 Regions and Spelling Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship V0.3.5 — add school-castle / home-cottage adventure regions, atomic two-step "medium" fill-letter (Elite/Boss only), MonsterKind→catalog visual mapping (Slime/Zombie/Dragon), region picker on HomePage with persistent preference, an explosion SFX for combo bursts, **and replace the shape-primitive avatars with bundled SVG character art (Slime / Zombie / Dragon / Magican)**.

**Architecture:** All six changes are orthogonal and ride on the V0.3 foundation. Question gains a third `kind` (`FillLetterMedium`) plus arrays for two-step state; `BattleEngine.submitAnswer` adds an "atomic step advance" branch that mutates `q.currentStep` in place without rotating to the next question or applying damage. `FillLetterGenerator.generateMedium` produces both blanks at construction; `PlanQuestionSource` routes Elite/Boss to medium with degrade-to-beginner-then-Choice fallbacks. Three regions share one `monsterPlan.slots` shape that resolves visuals through `monsterIndexForKind` (Slime / Zombie / Dragon). HomePage adds a chip-row region picker; selection persists via a tiny `TodayPreferences` (preferences-backed, mirroring `CoinAccount`/`WishlistStore`). For character art, four 1024×1024 SVGs land in `rawfile/character/`; `CharacterAssets` maps `MonsterKind` to a rawfile path; `CharacterCard` is rewritten from shape-primitive `@Builder`s to a single `Image($rawfile(this.assetPath))` rendering, while keeping all V0.2 hurt/cast animations on the wrapping `Stack`.

**Tech Stack:** HarmonyOS NEXT, ArkTS, ArkUI (`@Component`, `@State`, `@Builder`, `@Prop`, `ForEach`), `@ohos/hypium` for unit tests, `@kit.TestKit` (`Driver`, `ON`) for UI automation, `@ohos.data.preferences` for persistence, `hvigorw` build, `codelinter` static analysis, `hdc` device control.

```typescript
export enum QuestionKind {
  Choice = 'choice',
  FillLetter = 'fill-letter',
  FillLetterMedium = 'fill-letter-medium',
}

export class Question {
  promptZh: string = '';
  answer: string = '';
  options: string[] = [];
  wordId: string = '';

  kind: QuestionKind = QuestionKind.Choice;

  letterTemplate: string = '';
  missingIndex: number = -1;
  letterOptions: string[] = [];
  letterAnswer: string = '';

  letterTemplateBase: string = '';
  missingIndices: number[] = [];
  letterOptionsSteps: string[][] = [];
  letterAnswers: string[] = [];
  currentStep: number = 0;

  isValid(): boolean {
    if (this.wordId.length === 0) {
      return false;
    }
    if (this.answer.length === 0) {
      return false;
    }
    if (this.kind === QuestionKind.FillLetterMedium) {
      return this.isValidFillLetterMedium();
    }
    if (this.kind === QuestionKind.FillLetter) {
      return this.isValidFillLetter();
    }
    return this.isValidChoice();
  }

  // isValidChoice + isValidFillLetter unchanged

  private isValidFillLetterMedium(): boolean {
    if (this.letterTemplateBase.length === 0) {
      return false;
    }
    if (this.missingIndices.length !== 2) {
      return false;
    }
    if (this.letterOptionsSteps.length !== 2) {
      return false;
    }
    if (this.letterAnswers.length !== 2) {
      return false;
    }
    if (this.missingIndices[0] === this.missingIndices[1]) {
      return false;
    }
    if (this.currentStep < 0 || this.currentStep >= this.missingIndices.length) {
      return false;
    }
    const word: string = this.answer.toLowerCase();
    for (let i = 0; i < 2; i++) {
      const idx: number = this.missingIndices[i];
      if (idx < 1 || idx >= word.length) {
        return false;
      }
      const ans: string = this.letterAnswers[i];
      if (ans.length !== 1) {
        return false;
      }
      const opts: string[] = this.letterOptionsSteps[i];
      if (opts.length !== 3) {
        return false;
      }
      const seen: Set<string> = new Set<string>();
      for (let j = 0; j < opts.length; j++) {
        if (opts[j].length !== 1) {
          return false;
        }
        if (seen.has(opts[j])) {
          return false;
        }
        seen.add(opts[j]);
      }
      if (opts.indexOf(ans) < 0) {
        return false;
      }
      if (word.charAt(idx) !== ans) {
        return false;
      }
    }
    return true;
  }
}
```

```typescript
submitAnswer(option: string): AnswerOutcome {
  if (this.state_.status !== BattleStatus.Playing) {
    throw new Error(`BattleEngine.submitAnswer called while status=${this.state_.status}`);
  }
  const q: Question | undefined = this.state_.currentQuestion;
  if (q === undefined) {
    throw new Error('BattleEngine.submitAnswer called with no current question');
  }

  const isMedium: boolean = q.kind === QuestionKind.FillLetterMedium;
  const isFill: boolean = q.kind === QuestionKind.FillLetter;
  const validOptions: string[] = isMedium
    ? q.letterOptionsSteps[q.currentStep]
    : (isFill ? q.letterOptions : q.options);
  if (validOptions.indexOf(option) < 0) {
    throw new Error(`BattleEngine.submitAnswer: option "${option}" is not on the current question`);
  }

  const outcome: AnswerOutcome = new AnswerOutcome();
  const correct: boolean = isMedium
    ? option === q.letterAnswers[q.currentStep]
    : (isFill ? option === q.letterAnswer : option === q.answer);
  outcome.correct = correct;

  // Atomic medium: correct + non-final step advances in place; do NOT
  // touch combo / hp / counters / currentQuestion. The next submitAnswer
  // sees the same Question with currentStep + 1.
  if (isMedium && correct && q.currentStep < q.missingIndices.length - 1) {
    q.currentStep += 1;
    outcome.advancedStep = true;
    outcome.damage = 0;
    return outcome;
  }

  // Otherwise, fall through to the existing correct/wrong path.
  outcome.damage = 1;
  this.state_.totalAnswers += 1;
  if (correct) {
    this.state_.correctAnswers += 1;
    this.state_.comboCount += 1;
    if (this.state_.comboCount >= COMBO_BURST_THRESHOLD) {
      outcome.damage = COMBO_BURST_DAMAGE;
      outcome.comboTriggered = true;
      this.state_.comboCount = 0;
    }
    this.state_.monsterHp -= outcome.damage;

    if (this.state_.monsterHp <= 0) {
      this.state_.monsterHp = 0;
      this.state_.defeatedMonsters += 1;
      outcome.monsterDefeated = true;

      if (this.state_.defeatedMonsters >= this.state_.monstersTotal) {
        this.state_.status = BattleStatus.Won;
        this.state_.currentQuestion = undefined;
        outcome.battleEnded = true;
        outcome.endStatus = BattleStatus.Won;
        return outcome;
      }

      this.state_.monsterIndex += 1;
      this.state_.monsterHp = this.state_.monsterMaxHp;
      outcome.newMonsterSpawned = true;
    }
  } else {
    this.state_.comboCount = 0;
    this.state_.playerHp -= outcome.damage;
    if (this.state_.playerHp <= 0) {
      this.state_.playerHp = 0;
      this.state_.status = BattleStatus.Lost;
      this.state_.currentQuestion = undefined;
      outcome.battleEnded = true;
```

```typescript
generateMedium(word: WordEntry, _lastWordId?: string): Question | undefined {
  const lower: string = word.word.toLowerCase();
  const letters: string[] = [];
  for (let i = 0; i < lower.length; i++) {
    const c: string = lower.charAt(i);
    if (c >= 'a' && c <= 'z') {
      letters.push(c);
    }
  }
  if (letters.length < 4) {
    return undefined;
  }

  const idx0: number = this.randomInt(1, letters.length - 1);
  let idx1: number = this.randomInt(1, letters.length - 1);
  let attempts: number = 0;
  while (idx1 === idx0 && attempts < letters.length * 4) {
    idx1 = this.randomInt(1, letters.length - 1);
    attempts += 1;
  }
  if (idx1 === idx0) {
    return undefined;
  }

  const ans0: string = letters[idx0];
  const ans1: string = letters[idx1];

  const wordLetters: Set<string> = new Set<string>();
  for (let i = 0; i < letters.length; i++) {
    wordLetters.add(letters[i]);
  }
  const otherStepAnswer0: string = ans1;
  const otherStepAnswer1: string = ans0;

  const opts0: string[] = this.pickStepOptions(ans0, wordLetters, otherStepAnswer0);
  const opts1: string[] = this.pickStepOptions(ans1, wordLetters, otherStepAnswer1);

  const tmplParts: string[] = [];
  for (let i = 0; i < letters.length; i++) {
    if (i === idx0 || i === idx1) {
      tmplParts.push('_');
    } else {
      tmplParts.push(letters[i]);
    }
  }

  const q: Question = new Question();
  q.kind = QuestionKind.FillLetterMedium;
  q.promptZh = word.meaningZh;
  q.answer = word.word;
  q.options = [];
  q.wordId = word.id;
  q.letterTemplateBase = tmplParts.join(' ');
  q.missingIndices = [idx0, idx1];
  q.letterOptionsSteps = [opts0, opts1];
  q.letterAnswers = [ans0, ans1];
  q.currentStep = 0;
  return q;
}

private pickStepOptions(answer: string, wordLetters: Set<string>, otherAnswer: string): string[] {
  const candidatePool: string[] = [];
  for (let i = 0; i < ALPHABET.length; i++) {
    const c: string = ALPHABET.charAt(i);
    if (c === answer) continue;
    if (c === otherAnswer) continue;
    if (wordLetters.has(c)) continue;
    candidatePool.push(c);
  }
  if (candidatePool.length < TOTAL_OPTIONS - 1) {
    candidatePool.length = 0;
    for (let i = 0; i < ALPHABET.length; i++) {
      const c: string = ALPHABET.charAt(i);
      if (c !== answer) {
        candidatePool.push(c);
      }
    }
  }
  const shuffledPool: string[] = shuffle<string>(candidatePool, this.rng);
  const distractors: string[] = shuffledPool.slice(0, TOTAL_OPTIONS - 1);
  const optionsPlain: string[] = [answer, distractors[0], distractors[1]];
  return shuffle<string>(optionsPlain, this.rng);
}
```

```typescript
import preferences from '@ohos.data.preferences';
import { common } from '@kit.AbilityKit';
import { ADVENTURE_REGIONS, DEFAULT_REGION_ID } from '../data/AdventureCatalog';

const TODAY_PREFS_NAME: string = 'today_settings';
const KEY_REGION_ID: string = 'region_id';

export interface TodayPrefsHandle {
  get(key: string, def: string): Promise<string>;
  put(key: string, value: string): Promise<void>;
  flush(): Promise<void>;
}

class RealHandle implements TodayPrefsHandle {
  private readonly inner: preferences.Preferences;
  constructor(inner: preferences.Preferences) {
    this.inner = inner;
  }
  async get(key: string, def: string): Promise<string> {
    const v: preferences.ValueType = await this.inner.get(key, def);
    return typeof v === 'string' ? v : def;
  }
  async put(key: string, value: string): Promise<void> {
    await this.inner.put(key, value);
  }
  async flush(): Promise<void> {
    await this.inner.flush();
  }
}

export class TodayPreferences {
  private readonly handle: TodayPrefsHandle;

  constructor(handle: TodayPrefsHandle) {
    this.handle = handle;
  }

  static async open(ctx: common.UIAbilityContext): Promise<TodayPreferences> {
    const real: preferences.Preferences =
      await preferences.getPreferences(ctx, TODAY_PREFS_NAME);
    return new TodayPreferences(new RealHandle(real));
  }

  async loadRegionId(): Promise<string> {
    try {
      const id: string = await this.handle.get(KEY_REGION_ID, '');
      if (id.length === 0) {
        return DEFAULT_REGION_ID;
      }
      for (let i = 0; i < ADVENTURE_REGIONS.length; i++) {
        if (ADVENTURE_REGIONS[i].id === id) {
          return id;
        }
      }
      return DEFAULT_REGION_ID;
    } catch (e) {
      console.error(`TodayPreferences.loadRegionId failed: ${JSON.stringify(e)}`);
      return DEFAULT_REGION_ID;
    }
  }

  async saveRegionId(id: string): Promise<void> {
    try {
      await this.handle.put(KEY_REGION_ID, id);
      await this.handle.flush();
    } catch (e) {
      console.error(`TodayPreferences.saveRegionId failed: ${JSON.stringify(e)}`);
    }
  }
}

/** Test-only seam — bypass preferences with an in-memory handle. */
export class TodayPreferencesForTest {
  static withHandle(handle: TodayPrefsHandle): TodayPreferences {
    return new TodayPreferences(handle);
  }
}
```

```typescript
it('intermediateFillLetterAtomicFlow', 0, async (done: Function) => {
  const driver: Driver = await launchApp();
  await returnToHome(driver);

  const fruitChip = await driver.findComponent(ON.id('RegionChip_fruit-forest'));
  await fruitChip.click();
  await driver.delayMs(300);

  const cta = await driver.findComponent(ON.id('AdventureCardCta'));
  await cta.click();
  await driver.delayMs(1500);

  // Try each of the three option buttons in rotation up to 12 turns
  // total; check BattleFeedback text after each tap. The medium-step
  // success path is the only branch that emits "答对啦！下一空".
  const optionIds: string[] = ['BattleOptionA', 'BattleOptionB', 'BattleOptionC'];
  let stepAdvancedSeen: boolean = false;
  for (let turn = 0; turn < 12 && !stepAdvancedSeen; turn++) {
    const optId: string = optionIds[turn % 3];
    const btn = await driver.findComponent(ON.id(optId));
    if (btn === null) {
      break;
    }
    await btn.click();
    // Sample feedback inside the FEEDBACK_MS window (650 ms).
    await driver.delayMs(250);
    const feedback = await driver.findComponent(ON.id('BattleFeedback'));
    if (feedback !== null) {
      const text: string = await feedback.getText();
      if (text === '答对啦！下一空') {
        stepAdvancedSeen = true;
      }
    }
    // Wait the rest of the reveal window before next iteration.
    await driver.delayMs(550);
  }

  expect(stepAdvancedSeen).assertTrue();

  const finish = await driver.findComponent(ON.id('BattleFinishButton'));
  if (finish !== null) {
    await finish.click();
    await driver.delayMs(800);
  }
  await returnToHome(driver);

  done();
});
```
