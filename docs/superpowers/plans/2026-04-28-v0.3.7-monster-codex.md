# V0.3.7 Monster Codex Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a top-toolbar codex entry on `HomePage` and a new `MonsterCodexPage` that displays 3 archetype monster cards (Slime / Zombie / Dragon) with side-arrow navigation, lore-only Chinese descriptions, and no persistence/audio/swipe.

**Architecture:** Three-task surgery anchored on a single new data file (`MonsterCodex.ets`) that's deliberately decoupled from the battle `MonsterCatalog`. Task 1 adds the data + unit tests (TDD, no UI). Task 2 adds the page + route registration (UI compile + codelinter only — no UI tests yet because there's no entry button). Task 3 adds the toolbar entry button on `HomePage` and the 4-case UI test suite (TDD: tests written and made to fail before the entry button exists, then made to pass after the button is wired). Each task is independently buildable and revertible.

**Tech Stack:** ArkTS / HarmonyOS NEXT; `@kit.TestKit` (Driver / ON / Component) for UI tests; `@ohos/hypium` for unit + UI assertions; `@Builder` inline composition (codelinter `avoid-overusing-custom-component-check`); `router.pushUrl` / `router.back` for navigation; `Image($rawfile(...))` with `syncLoad(true)` for test-stable rendering.

**Spec:** [docs/superpowers/specs/2026-04-28-v0.3.7-monster-codex-design.md](../specs/2026-04-28-v0.3.7-monster-codex-design.md)

```typescript
import { router } from '@kit.ArkUI';
import { CodexEntry, MONSTER_CODEX } from '../data/MonsterCodex';

/**
 * MonsterCodexPage: read-only browse of 3 archetype monsters
 * (Slime / Zombie / Dragon). No persistence — every entry starts
 * at index 0. No audio. No swipe / long-press / zoom — only the
 * left/right circular arrow buttons drive navigation, and they
 * disable at boundaries (no wrap-around).
 *
 * ArkUI reactivity note (lesson from v0.3.6 letter-spacing fix):
 * every @Builder below is parameter-less and reads `this.currentIndex`
 * directly (or via the helper methods `current()` / `hasPrev()` /
 * `hasNext()`). Do NOT pass parameters into the @Builder bodies —
 * ArkUI's compile-time dependency analysis only tracks @State reads
 * inside the Builder body's source, so a parameterised Builder will
 * silently freeze on the initial frame.
 */
@Entry
@Component
struct MonsterCodexPage {
  @State private currentIndex: number = 0;

  private hasPrev(): boolean {
    return this.currentIndex > 0;
  }

  private hasNext(): boolean {
    return this.currentIndex < MONSTER_CODEX.length - 1;
  }

  private current(): CodexEntry {
    return MONSTER_CODEX[this.currentIndex];
  }

  private goPrev(): void {
    if (this.hasPrev()) {
      this.currentIndex -= 1;
    }
  }

  private goNext(): void {
    if (this.hasNext()) {
      this.currentIndex += 1;
    }
  }

  build() {
    Column() {
      this.topBar();
      this.avatarRow();
      this.nameBlock();
      this.descriptionBlock();
      this.positionIndicator();
    }
    .width('100%')
    .height('100%')
    .padding({ top: 16, left: 16, right: 16, bottom: 32 })
    .backgroundColor('#F8F9FA');
  }

  @Builder
  private topBar() {
    Row() {
      Button('←')
        .id('CodexBackButton')
        .type(ButtonType.Circle)
        .width(40)
        .height(40)
        .fontSize(20)
        .fontColor('#1D3557')
        .backgroundColor('#FFFFFF')
        .onClick((): void => {
          router.back();
        });

      Text('怪物图鉴')
        .fontSize(20)
        .fontWeight(FontWeight.Bold)
        .fontColor('#1D3557')
        .layoutWeight(1)
        .textAlign(TextAlign.Center)
        .margin({ right: 40 });
    }
    .width('100%');
  }

  @Builder
  private avatarRow() {
    Row({ space: 16 }) {
      Button('⬅')
        .id('CodexPrev')
        .type(ButtonType.Circle)
        .width(56)
        .height(56)
        .fontSize(28)
        .fontColor(this.hasPrev() ? '#E63946' : '#999999')
        .backgroundColor(this.hasPrev() ? '#FCEAEA' : '#EEEEEE')
        .enabled(this.hasPrev())
        .onClick((): void => {
          this.goPrev();
        });

      Stack() {
        Column()
          .width(160)
          .height(160)
          .backgroundColor('#FFFFFF')
          .borderRadius(24)
          .border({ width: 1, color: '#E0E0E0' });

        Image($rawfile(this.current().assetPath))
          .id('CodexAvatar')
          .objectFit(ImageFit.Contain)
          .width(128)
          .height(128)
          .syncLoad(true)
          .interpolation(ImageInterpolation.High);
      }
      .width(160)
      .height(160)
      .alignContent(Alignment.Center);

      Button('➡')
        .id('CodexNext')
        .type(ButtonType.Circle)
        .width(56)
        .height(56)
        .fontSize(28)
        .fontColor(this.hasNext() ? '#E63946' : '#999999')
        .backgroundColor(this.hasNext() ? '#FCEAEA' : '#EEEEEE')
        .enabled(this.hasNext())
        .onClick((): void => {
          this.goNext();
        });
    }
    .width('100%')
    .justifyContent(FlexAlign.Center)
    .margin({ top: 32 });
  }

  @Builder
  private nameBlock() {
    Column({ space: 8 }) {
      Text(this.current().nameEn)
        .id('CodexName')
        .fontSize(22)
        .fontWeight(FontWeight.Bold)
        .fontColor('#1D3557');

      Text(`「${this.current().kindLabelZh}」`)
        .id('CodexKindLabel')
        .fontSize(14)
        .fontColor('#457B9D')
        .backgroundColor('#EAF2F8')
        .padding({ left: 12, right: 12, top: 4, bottom: 4 })
        .borderRadius(12);
    }
    .alignItems(HorizontalAlign.Center)
    .margin({ top: 24 });
  }

  @Builder
  private descriptionBlock() {
    Text(this.current().descriptionZh)
      .id('CodexDescription')
      .fontSize(14)
      .lineHeight(22)
      .fontColor('#333333')
      .textAlign(TextAlign.Center)
      .padding({ left: 24, right: 24 })
      .margin({ top: 24 });
  }

  @Builder
  private positionIndicator() {
    Text(`${this.currentIndex + 1} / ${MONSTER_CODEX.length}`)
      .id('CodexPositionIndicator')
      .fontSize(12)
      .fontColor('#888888')
      .margin({ top: 24 });
  }
}
```

Create `entry/src/ohosTest/ets/test/MonsterCodexFlow.ui.test.ets`:

```typescript
import { describe, it, expect } from '@ohos/hypium';
import { abilityDelegatorRegistry, Driver, ON, Component } from '@kit.TestKit';
import { Want } from '@kit.AbilityKit';
import { clickByIdShared } from './RoutingFlow.ui.test';

const BUNDLE: string = 'com.terryma.wordmagicgame';
const DELEGATOR: abilityDelegatorRegistry.AbilityDelegator =
  abilityDelegatorRegistry.getAbilityDelegator();

async function launchApp(): Promise<Driver> {
  const want: Want = { bundleName: BUNDLE, abilityName: 'EntryAbility' };
  await DELEGATOR.startAbility(want);
  const driver: Driver = Driver.create();
  await driver.delayMs(1000);
  return driver;
}

/**
 * Mirrors the helper in HomeToolbarLocked.ui.test — press back up to
 * 5 times until HomeStartButton resolves so each `it` block starts
 * on the HomePage even if a previous suite left the app mid-flow.
 */
async function returnToHome(driver: Driver): Promise<void> {
  for (let i: number = 0; i < 5; i++) {
    const home: Component | null =
      await driver.findComponent(ON.id('HomeStartButton'));
    if (home !== null) {
      return;
    }
    await driver.pressBack();
    await driver.delayMs(500);
  }
}

async function openCodex(driver: Driver): Promise<void> {
  await clickByIdShared(driver, 'HomeCodexButton');
  await driver.delayMs(1500);
}

/**
 * V0.3.7 acceptance:
 *  - HomePage toolbar surfaces 📖 HomeCodexButton (always enabled).
 *  - Codex page enters at index 0 (Slime), shows '1 / 3' indicator,
 *    CodexPrev disabled, CodexNext enabled.
 *  - Clicking CodexNext walks Slime → Zombie → Dragon and disables at
 *    end; clicking CodexPrev walks back and disables at start. Extra
 *    boundary clicks are no-ops (do not crash, do not wrap).
 *  - CodexBackButton returns to HomePage; HomeStartButton is reachable.
 */
export default function monsterCodexFlowUiTest(): void {
  describe('MonsterCodexFlow', () => {
    it('homeCodexButtonOpensCodexPage', 0, async (done: Function) => {
      const driver: Driver = await launchApp();
      await returnToHome(driver);
      await openCodex(driver);
      await driver.assertComponentExist(ON.id('CodexAvatar'));
      await driver.assertComponentExist(ON.text('Slime'));
      await driver.assertComponentExist(ON.text('「普通怪物」'));
      await driver.assertComponentExist(ON.text('1 / 3'));
      done();
    });

    it('nextArrowAdvancesAndDisablesAtEnd', 0, async (done: Function) => {
      const driver: Driver = await launchApp();
      await returnToHome(driver);
      await openCodex(driver);
      await clickByIdShared(driver, 'CodexNext');
      await driver.delayMs(300);
      await driver.assertComponentExist(ON.text('Zombie'));
      await driver.assertComponentExist(ON.text('2 / 3'));
      await clickByIdShared(driver, 'CodexNext');
      await driver.delayMs(300);
      await driver.assertComponentExist(ON.text('Dragon'));
      await driver.assertComponentExist(ON.text('3 / 3'));
      // Boundary: extra tap on disabled CodexNext is a no-op.
      await clickByIdShared(driver, 'CodexNext');
      await driver.delayMs(300);
      await driver.assertComponentExist(ON.text('Dragon'));
      await driver.assertComponentExist(ON.text('3 / 3'));
      done();
    });

    it('prevArrowGoesBackAndDisablesAtStart', 0, async (done: Function) => {
      const driver: Driver = await launchApp();
      await returnToHome(driver);
      await openCodex(driver);
      await clickByIdShared(driver, 'CodexNext');
      await driver.delayMs(300);
      await clickByIdShared(driver, 'CodexNext');
      await driver.delayMs(300);
      await driver.assertComponentExist(ON.text('Dragon'));
      await clickByIdShared(driver, 'CodexPrev');
      await driver.delayMs(300);
      await driver.assertComponentExist(ON.text('Zombie'));
      await clickByIdShared(driver, 'CodexPrev');
      await driver.delayMs(300);
      await driver.assertComponentExist(ON.text('Slime'));
      await driver.assertComponentExist(ON.text('1 / 3'));
      // Boundary: extra tap on disabled CodexPrev is a no-op.
      await clickByIdShared(driver, 'CodexPrev');
      await driver.delayMs(300);
      await driver.assertComponentExist(ON.text('Slime'));
      done();
    });

    it('backButtonReturnsToHome', 0, async (done: Function) => {
      const driver: Driver = await launchApp();
      await returnToHome(driver);
      await openCodex(driver);
      await driver.assertComponentExist(ON.id('CodexAvatar'));
      await clickByIdShared(driver, 'CodexBackButton');
      await driver.delayMs(1500);
      await driver.assertComponentExist(ON.id('HomeStartButton'));
      const avatar: Component | null =
        await driver.findComponent(ON.id('CodexAvatar'));
      expect(avatar === null).assertTrue();
      done();
    });
  });
}
```
