# WordMagicGame V0.3 Fun Learning Core Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Turn the existing battle prototype into the V0.3 "fun-learning core": today's adventure entry, fruit-forest themed region, monster-driven question kinds (choice + fill-letter), light memory state machine, magic coin account with daily cap, and a 3-item wishlist with parent long-press confirmation.

**Architecture:** Layer the new logic strictly under existing patterns. Question gains a `kind` discriminator + optional fill-letter fields (no inheritance); `BattleEngine` accepts an `IQuestionSource` interface so it can run from either the existing `QuestionGenerator` (free-practice / review) or a new `PlanQuestionSource` (today's adventure). `LearningRecorder` delegates state-machine math to a pure `MemoryScheduler`; `LearningSnapshot` schema bumps to v2 with backward-compat parsing. New singletons `CoinAccount` and `WishlistStore` mirror the recorder's preferences pattern. ArkUI gains one new page (`WishlistPage`), one container component (`AdventureCard`), one input component (`ParentLongPressGate`), and one display component (`LetterTemplate`) — battle scene reuses `ChoiceButton` for both question kinds.

**Tech Stack:** HarmonyOS NEXT, ArkTS, ArkUI components (`@Component`, `@Prop`, `@State`, `@Watch`, `@Builder`, `setInterval`), `@ohos/hypium` for unit tests, `@kit.TestKit` (`Driver`, `ON`) for UI automation, `@ohos.data.preferences` for persistence, `hvigorw` build, `codelinter` static analysis, `hdc` device control.

**Spec:** [`docs/superpowers/specs/2026-04-28-v0.3-fun-learning-core-design.md`](../specs/2026-04-28-v0.3-fun-learning-core-design.md)

- **Pass signal:** test report ends with `OK` and the new `it(name, …)` lines appear in the run summary.

```typescript
/**
 * Question — one prompt shown to the player (spec §2.1).
 *
 * V0.3 introduces a `kind` discriminator. Choice questions keep
 * the V0.1/V0.2 invariants exactly; FillLetter questions add a
 * letter template + 3 candidate letters and reuse `answer` to
 * carry the full English word (TTS auto-speak still works).
 */
export enum QuestionKind {
  Choice = 'choice',
  FillLetter = 'fill-letter',
}

export class Question {
  promptZh: string = '';
  answer: string = '';
  options: string[] = [];
  wordId: string = '';

  kind: QuestionKind = QuestionKind.Choice;

  letterTemplate: string = '';
  missingIndex: number = -1;
  letterOptions: string[] = [];
  letterAnswer: string = '';

  isValid(): boolean {
    if (this.wordId.length === 0) {
      return false;
    }
    if (this.answer.length === 0) {
      return false;
    }
    if (this.kind === QuestionKind.FillLetter) {
      return this.isValidFillLetter();
    }
    return this.isValidChoice();
  }

  private isValidChoice(): boolean {
    if (this.promptZh.length === 0) {
      return false;
    }
    if (this.options.length !== 3) {
      return false;
    }
    if (this.options.indexOf(this.answer) < 0) {
      return false;
    }
    const seen: Set<string> = new Set<string>();
    for (let i = 0; i < this.options.length; i++) {
      if (seen.has(this.options[i])) {
        return false;
      }
      seen.add(this.options[i]);
    }
    return true;
  }

  private isValidFillLetter(): boolean {
    if (this.letterTemplate.length === 0) {
      return false;
    }
    if (this.letterOptions.length !== 3) {
      return false;
    }
    const seen: Set<string> = new Set<string>();
    for (let i = 0; i < this.letterOptions.length; i++) {
      const c: string = this.letterOptions[i];
      if (c.length !== 1) {
        return false;
      }
      if (seen.has(c)) {
        return false;
      }
      seen.add(c);
    }
    if (this.letterOptions.indexOf(this.letterAnswer) < 0) {
      return false;
    }
    if (this.missingIndex < 0) {
      return false;
    }
    const word: string = this.answer.toLowerCase();
    if (this.missingIndex >= word.length) {
      return false;
    }
    if (word.charAt(this.missingIndex) !== this.letterAnswer) {
      return false;
    }
    return true;
  }
}
```

```typescript
import { WordEntry } from '../models/WordEntry';
import { Question, QuestionKind } from '../models/Question';
import { RandomFn, shuffle } from '../utils/shuffle';

const MIN_WORD_LETTERS: number = 3;
const ALPHABET: string = 'abcdefghijklmnopqrstuvwxyz';
const TOTAL_OPTIONS: number = 3;

/**
 * Generate a beginner fill-letter question (one missing letter,
 * three candidates) from a single WordEntry. Returns undefined
 * for words with fewer than three alphabetic characters so the
 * caller can degrade to a Choice question.
 *
 * Determinism: every random pick reads from `rng` so the same
 * seed produces the same Question.
 */
export class FillLetterGenerator {
  private readonly rng: RandomFn;

  constructor(rng?: RandomFn) {
    this.rng = rng ?? Math.random;
  }

  generate(word: WordEntry, _lastWordId?: string): Question | undefined {
    const lower: string = word.word.toLowerCase();
    const letters: string[] = [];
    for (let i = 0; i < lower.length; i++) {
      const c: string = lower.charAt(i);
      if (c >= 'a' && c <= 'z') {
        letters.push(c);
      }
    }
    if (letters.length < MIN_WORD_LETTERS) {
      return undefined;
    }

    // Pick a missing index from [1, letters.length - 1] so the
    // first letter is never hidden — easier for early readers.
    const missingIndex: number = this.randomInt(1, letters.length - 1);
    const letterAnswer: string = letters[missingIndex];

    // Distractors: random letters not in the word at all (and not
    // equal to letterAnswer). Falls back to "letters not equal to
    // letterAnswer" if the word covers most of the alphabet.
    const wordLetters: Set<string> = new Set<string>();
    for (let i = 0; i < letters.length; i++) {
      wordLetters.add(letters[i]);
    }
    const candidatePool: string[] = [];
    for (let i = 0; i < ALPHABET.length; i++) {
      const c: string = ALPHABET.charAt(i);
      if (wordLetters.has(c)) {
        continue;
      }
      candidatePool.push(c);
    }
    if (candidatePool.length < TOTAL_OPTIONS - 1) {
      // Edge: word covers most letters. Fall back to alphabet
      // minus letterAnswer so we still produce a question.
      candidatePool.length = 0;
      for (let i = 0; i < ALPHABET.length; i++) {
        const c: string = ALPHABET.charAt(i);
        if (c !== letterAnswer) {
          candidatePool.push(c);
        }
      }
    }
    const shuffledPool: string[] = shuffle<string>(candidatePool, this.rng);
    const distractors: string[] = shuffledPool.slice(0, TOTAL_OPTIONS - 1);

    const optionsPlain: string[] = [letterAnswer, distractors[0], distractors[1]];
    const letterOptions: string[] = shuffle<string>(optionsPlain, this.rng);

    const templateParts: string[] = [];
    for (let i = 0; i < letters.length; i++) {
      templateParts.push(i === missingIndex ? '_' : letters[i]);
    }

    const q: Question = new Question();
    q.kind = QuestionKind.FillLetter;
    q.promptZh = word.meaningZh;
    q.answer = word.word;
    q.options = [];
    q.wordId = word.id;
    q.letterTemplate = templateParts.join(' ');
    q.missingIndex = missingIndex;
    q.letterOptions = letterOptions;
    q.letterAnswer = letterAnswer;
    return q;
  }

  private randomInt(minInclusive: number, maxInclusive: number): number {
    if (maxInclusive < minInclusive) {
      return minInclusive;
    }
    const span: number = maxInclusive - minInclusive + 1;
    const r: number = this.rng();
    const idx: number = Math.floor(r * span);
    if (idx < 0) {
      return minInclusive;
    }
    if (idx >= span) {
      return maxInclusive;
    }
    return minInclusive + idx;
  }
}
```

```typescript
const SLOT_GAP: number = 8;
const SLOT_WIDTH: number = 36;
const SLOT_HEIGHT: number = 48;
const TEXT_COLOR_FILLED: string = '#1D3557';
const TEXT_COLOR_MISSING: string = '#E63946';
const UNDERLINE_COLOR: string = '#E63946';
const BG_COLOR_MISSING: string = '#FCEAEA';

/**
 * LetterTemplate — display-only row of letters, with the missing
 * slot rendered as a coloured `_` over a faint pink background.
 *
 * Consumers pass the space-separated template ("a _ p l e") and the
 * 0-based missingIndex; the component splits and renders each token.
 */
@Component
export struct LetterTemplate {
  @Prop template: string = '';
  @Prop missingIndex: number = -1;

  build(): void {
    Row({ space: SLOT_GAP }) {
      ForEach(this.tokens(), (item: LetterSlot, _: number) => {
        Stack() {
          if (item.isMissing) {
            Column()
              .width(SLOT_WIDTH)
              .height(SLOT_HEIGHT)
              .backgroundColor(BG_COLOR_MISSING)
              .borderRadius(6);
            Text('_')
              .fontSize(28)
              .fontWeight(FontWeight.Bold)
              .fontColor(TEXT_COLOR_MISSING);
          } else {
            Text(item.glyph)
              .fontSize(32)
              .fontWeight(FontWeight.Bold)
              .fontColor(TEXT_COLOR_FILLED);
          }
        }
        .width(SLOT_WIDTH)
        .height(SLOT_HEIGHT)
        .alignContent(Alignment.Center);
      }, (item: LetterSlot, _: number) => `${item.idx}-${item.glyph}`);
    }
    .id('LetterTemplateRow')
    .width('100%')
    .justifyContent(FlexAlign.Center);
  }

  private tokens(): LetterSlot[] {
    const out: LetterSlot[] = [];
    if (this.template.length === 0) {
      return out;
    }
    const parts: string[] = this.template.split(' ');
    for (let i = 0; i < parts.length; i++) {
      const slot: LetterSlot = new LetterSlot();
      slot.idx = i;
      slot.glyph = parts[i];
      slot.isMissing = i === this.missingIndex || parts[i] === '_';
      out.push(slot);
    }
    return out;
  }
}

class LetterSlot {
  idx: number = 0;
  glyph: string = '';
  isMissing: boolean = false;
}
```

```typescript
import { WordStat } from './WrongAnswerStore';

export enum MemoryState {
  New = 'new',
  Learning = 'learning',
  Review = 'review',
  Familiar = 'familiar',
  Mastered = 'mastered',
}

export const INTERVAL_30_MIN_MS: number = 30 * 60 * 1000;
export const INTERVAL_1_DAY_MS: number = 24 * 3600 * 1000;
export const INTERVAL_3_DAY_MS: number = 3 * INTERVAL_1_DAY_MS;
export const INTERVAL_14_DAY_MS: number = 14 * INTERVAL_1_DAY_MS;

const PROMOTE_TO_FAMILIAR_AT: number = 2;
const PROMOTE_TO_MASTERED_AT: number = 5;

/**
 * Pure state-machine that decides the next memoryState +
 * nextReviewMs after a single answer. LearningRecorder feeds this
 * with the WordStat after it has bumped seen/correct/wrong, so
 * the scheduler can read the latest correctCount when picking
 * promotion thresholds.
 */
export class MemoryScheduler {
  applyAnswer(stat: WordStat, correct: boolean, nowMs: number): void {
    const current: MemoryState = this.coerceState(stat.memoryState);
    if (correct) {
      stat.lastCorrectMs = nowMs;
      this.applyCorrect(stat, current, nowMs);
    } else {
      this.applyWrong(stat, nowMs);
    }
  }

  classify(stat: WordStat, nowMs: number): MemoryState {
    if (stat.nextReviewMs > 0 && stat.nextReviewMs <= nowMs) {
      return MemoryState.Review;
    }
    return this.coerceState(stat.memoryState);
  }

  private applyCorrect(stat: WordStat, current: MemoryState, nowMs: number): void {
    if (current === MemoryState.New) {
      stat.memoryState = MemoryState.Learning;
      stat.nextReviewMs = nowMs + INTERVAL_1_DAY_MS;
      return;
    }
    if (current === MemoryState.Learning) {
      if (stat.correctCount >= PROMOTE_TO_FAMILIAR_AT) {
        stat.memoryState = MemoryState.Familiar;
        stat.nextReviewMs = nowMs + INTERVAL_3_DAY_MS;
      } else {
        stat.nextReviewMs = nowMs + INTERVAL_1_DAY_MS;
      }
      return;
    }
    if (current === MemoryState.Review) {
      stat.memoryState = MemoryState.Learning;
      stat.nextReviewMs = nowMs + INTERVAL_1_DAY_MS;
      return;
    }
    if (current === MemoryState.Familiar) {
      if (stat.correctCount >= PROMOTE_TO_MASTERED_AT) {
        stat.memoryState = MemoryState.Mastered;
        stat.nextReviewMs = nowMs + INTERVAL_14_DAY_MS;
      } else {
        stat.nextReviewMs = nowMs + INTERVAL_3_DAY_MS;
      }
      return;
    }
    // Mastered
    stat.nextReviewMs = nowMs + INTERVAL_14_DAY_MS;
  }

  private applyWrong(stat: WordStat, nowMs: number): void {
    stat.memoryState = MemoryState.Review;
    stat.nextReviewMs = nowMs + INTERVAL_30_MIN_MS;
  }

  private coerceState(label: string): MemoryState {
    if (label === MemoryState.Learning) return MemoryState.Learning;
    if (label === MemoryState.Review) return MemoryState.Review;
    if (label === MemoryState.Familiar) return MemoryState.Familiar;
    if (label === MemoryState.Mastered) return MemoryState.Mastered;
    return MemoryState.New;
  }
}
```

```typescript
   * Return wordIds whose current MemoryState (with due-check) matches `state`,
   * most-recent-first by lastAnsweredMs. New (no stat) words are included
   * only when state === MemoryState.New is requested AND the caller has a
   * separate way to enumerate the universe — recorder cannot list every
   * unseen wordId, so this list reflects only known words.

  wordIdsByMemoryState(state: MemoryState, limit: number, nowMs: number): string[] {
    if (limit <= 0) {
      return [];
    }
    const out: string[] = [];
    const stats: WordStat[] = this.snapshot.stats.slice();
    stats.sort((a: WordStat, b: WordStat): number => b.lastAnsweredMs - a.lastAnsweredMs);
    for (let i = 0; i < stats.length && out.length < limit; i++) {
      if (this.scheduler.classify(stats[i], nowMs) === state) {
        out.push(stats[i].wordId);
      }
    }
    return out;
  }
```

```typescript
import { WordEntry } from '../models/WordEntry';
import { WordRepository } from './WordRepository';
import { LearningRecorder } from './LearningRecorder';
import { MemoryScheduler, MemoryState } from './MemoryScheduler';
import { WordStat } from './WrongAnswerStore';
import {
  AdventureRegion,
  TodaySessionPlan,
  WordSlot,
  WordSource,
  MONSTER_PLAN_SLOT_COUNT,
  WORD_PLAN_MULTIPLIER,
} from '../models/AdventureRegion';
import { RandomFn, shuffle } from '../utils/shuffle';

const RATIO_REVIEW: number = 0.5;
const RATIO_LEARNING: number = 0.3;
// 'new' = remainder
const BOSS_WORDS_LIMIT: number = MONSTER_PLAN_SLOT_COUNT;
const MIN_TOTAL_WORDS: number = MONSTER_PLAN_SLOT_COUNT;

/**
 * Build a TodaySessionPlan: pick 50% review / 30% learning / 20% new
 * words from the universe, weighted toward the region's theme
 * categories. The plan is consumed by PlanQuestionSource at battle time.
 */
export class TodayAdventureBuilder {
  private readonly repo: WordRepository;
  private readonly recorder: LearningRecorder;
  private readonly scheduler: MemoryScheduler;
  private readonly rng: RandomFn;

  constructor(
    repo: WordRepository,
    recorder: LearningRecorder,
    scheduler: MemoryScheduler,
    rng?: RandomFn
  ) {
    this.repo = repo;
    this.recorder = recorder;
    this.scheduler = scheduler;
    this.rng = rng ?? Math.random;
  }

  build(region: AdventureRegion, isFirstToday: boolean, nowMs: number): TodaySessionPlan {
    const universe: WordEntry[] = this.repo.all();
    if (universe.length < MIN_TOTAL_WORDS) {
      throw new Error(`TodayAdventureBuilder: need at least ${MIN_TOTAL_WORDS} words, got ${universe.length}`);
    }

    // Classify and bucket.
    const reviewBucket: WordEntry[] = [];
    const learningBucket: WordEntry[] = [];
    const newBucket: WordEntry[] = [];
    const statById: Map<string, WordStat> = this.indexStatsByWordId();
    for (let i = 0; i < universe.length; i++) {
      const w: WordEntry = universe[i];
      const stat: WordStat | undefined = statById.get(w.id);
      let state: MemoryState;
      if (stat === undefined) {
        state = MemoryState.New;
      } else {
        state = this.scheduler.classify(stat, nowMs);
      }
      if (state === MemoryState.Review) {
        reviewBucket.push(w);
      } else if (state === MemoryState.New) {
        newBucket.push(w);
      } else {
        learningBucket.push(w);
      }
    }

    // Theme-weight then shuffle each bucket.
    const themed: Set<string> = new Set<string>();
    for (let i = 0; i < region.themeWordCategories.length; i++) {
      themed.add(region.themeWordCategories[i]);
    }
    const reviewShuffled: WordEntry[] = this.weightShuffle(reviewBucket, themed);
    const learningShuffled: WordEntry[] = this.weightShuffle(learningBucket, themed);
    const newShuffled: WordEntry[] = this.weightShuffle(newBucket, themed);

    const wordSlotCount: number = MONSTER_PLAN_SLOT_COUNT * WORD_PLAN_MULTIPLIER;
    const wantReview: number = Math.round(wordSlotCount * RATIO_REVIEW);
    const wantLearning: number = Math.round(wordSlotCount * RATIO_LEARNING);
    const wantNew: number = wordSlotCount - wantReview - wantLearning;

    const plan: TodaySessionPlan = new TodaySessionPlan();
    plan.regionId = region.id;
    plan.monsterSlots = region.monsterPlan.slots.slice();
    plan.isFirstToday = isFirstToday;
    plan.wordPlan = [];

    this.takeFromBucket(plan.wordPlan, reviewShuffled, wantReview, WordSource.Review);
    this.takeFromBucket(plan.wordPlan, learningShuffled, wantLearning, WordSource.Learning);
    this.takeFromBucket(plan.wordPlan, newShuffled, wantNew, WordSource.New);

    // Backfill in priority order if any bucket ran short.
    while (plan.wordPlan.length < wordSlotCount) {
      const before: number = plan.wordPlan.length;
      this.backfill(plan.wordPlan, reviewShuffled, WordSource.Review, 1);
      if (plan.wordPlan.length === before) {
        this.backfill(plan.wordPlan, learningShuffled, WordSource.Learning, 1);
      }
      if (plan.wordPlan.length === before) {
        this.backfill(plan.wordPlan, newShuffled, WordSource.New, 1);
      }
      if (plan.wordPlan.length === before) {
        break;
      }
    }

    plan.bossWordIds = this.computeBossWordIds(plan.wordPlan);
    return plan;
  }

  private indexStatsByWordId(): Map<string, WordStat> {
    const out: Map<string, WordStat> = new Map<string, WordStat>();
    const stats: WordStat[] = this.recorder.statsForTest();
    for (let i = 0; i < stats.length; i++) {
      out.set(stats[i].wordId, stats[i]);
    }
    return out;
  }

  private weightShuffle(items: WordEntry[], themed: Set<string>): WordEntry[] {
    if (items.length === 0) {
      return [];
    }
    const themedItems: WordEntry[] = [];
    const otherItems: WordEntry[] = [];
    for (let i = 0; i < items.length; i++) {
      if (themed.has(items[i].category)) {
        themedItems.push(items[i]);
      } else {
        otherItems.push(items[i]);
      }
    }
    const shuffledThemed: WordEntry[] = shuffle<WordEntry>(themedItems, this.rng);
    const shuffledOther: WordEntry[] = shuffle<WordEntry>(otherItems, this.rng);
    const interleaved: WordEntry[] = [];
    // Themed first, with double weight: alternate themed,themed,other.
    let ti: number = 0;
    let oi: number = 0;
    while (ti < shuffledThemed.length || oi < shuffledOther.length) {
      if (ti < shuffledThemed.length) {
        interleaved.push(shuffledThemed[ti++]);
      }
      if (ti < shuffledThemed.length) {
        interleaved.push(shuffledThemed[ti++]);
      }
      if (oi < shuffledOther.length) {
        interleaved.push(shuffledOther[oi++]);
      }
    }
    return interleaved;
  }

  private takeFromBucket(out: WordSlot[], bucket: WordEntry[], n: number, source: WordSource): void {
    while (n > 0 && bucket.length > 0) {
      const w: WordEntry = bucket.shift() as WordEntry;
      const slot: WordSlot = new WordSlot();
      slot.wordId = w.id;
      slot.source = source;
      out.push(slot);
      n -= 1;
    }
  }

  private backfill(out: WordSlot[], bucket: WordEntry[], source: WordSource, n: number): void {
    this.takeFromBucket(out, bucket, n, source);
  }

  private computeBossWordIds(wordPlan: WordSlot[]): string[] {
    // Sort by source priority (review > learning > new), then keep first BOSS_WORDS_LIMIT distinct.
    const priority: Map<string, number> = new Map<string, number>();
    priority.set(WordSource.Review, 0);
    priority.set(WordSource.Learning, 1);
    priority.set(WordSource.New, 2);
    const sorted: WordSlot[] = wordPlan.slice();
    sorted.sort((a: WordSlot, b: WordSlot): number => {
      const pa: number = priority.has(a.source) ? (priority.get(a.source) as number) : 9;
      const pb: number = priority.has(b.source) ? (priority.get(b.source) as number) : 9;
      return pa - pb;
    });
    const seen: Set<string> = new Set<string>();
    const out: string[] = [];
    for (let i = 0; i < sorted.length && out.length < BOSS_WORDS_LIMIT; i++) {
      if (seen.has(sorted[i].wordId)) {
        continue;
      }
      seen.add(sorted[i].wordId);
      out.push(sorted[i].wordId);
    }
    return out;
  }
}
```

```typescript
import { Question, QuestionKind } from '../models/Question';
import { WordEntry } from '../models/WordEntry';
import { WordRepository } from './WordRepository';
import { QuestionGenerator } from './QuestionGenerator';
import { FillLetterGenerator } from './FillLetterGenerator';
import { IQuestionSource } from './IQuestionSource';
import {
  TodaySessionPlan,
  MonsterSlot,
  WordSlot,
} from '../models/AdventureRegion';
import { MonsterKind } from '../data/MonsterCatalog';
import { RandomFn } from '../utils/shuffle';

type MonsterIndexProvider = () => number;

const DEFAULT_PROVIDER: MonsterIndexProvider = (): number => 1;

/**
 * Question source for today's adventure. Consumes the plan
 * one slot at a time; reads the current 1-based monster index
 * from an injected provider so BattleEngine.advance() never
 * has to know about the source.
 */
export class PlanQuestionSource implements IQuestionSource {
  private readonly plan: TodaySessionPlan;
  private readonly repo: WordRepository;
  private readonly fillGen: FillLetterGenerator;
  private readonly choiceGen: QuestionGenerator;
  private readonly rng: RandomFn;
  private monsterIndexProvider: MonsterIndexProvider = DEFAULT_PROVIDER;
  private wordCursor: number = 0;
  // Local mutable copy so we can shift() without mutating the plan param.
  private bossWordIdsLocal: string[];

  constructor(
    plan: TodaySessionPlan,
    repo: WordRepository,
    fillGen: FillLetterGenerator,
    choiceGen: QuestionGenerator,
    rng?: RandomFn
  ) {
    this.plan = plan;
    this.repo = repo;
    this.fillGen = fillGen;
    this.choiceGen = choiceGen;
    this.rng = rng ?? Math.random;
    this.bossWordIdsLocal = plan.bossWordIds.slice();
  }

  setMonsterIndexProvider(provider: MonsterIndexProvider): void {
    this.monsterIndexProvider = provider;
  }

  nextQuestion(lastWordId?: string): Question {
    const idx1Based: number = this.monsterIndexProvider();
    const idx0: number = this.clampSlot(idx1Based - 1);
    const slot: MonsterSlot = this.plan.monsterSlots[idx0];

    const desiredKind: QuestionKind = this.kindForSlot(slot.kind);
    const wordId: string = this.pickWordId(slot.kind, lastWordId);
    const entry: WordEntry | undefined = this.repo.findById(wordId);

    if (entry === undefined) {
      // Fallback: produce a generic Choice question from the repo.
      return this.choiceGen.nextQuestion(lastWordId);
    }

    if (desiredKind === QuestionKind.FillLetter) {
      const q: Question | undefined = this.fillGen.generate(entry, lastWordId);
      if (q !== undefined) {
        return q;
      }
      // Short-word fallback to Choice.
    }
    return this.choiceGen.nextQuestionForWord(entry, lastWordId);
  }

  private clampSlot(idx0: number): number {
    if (idx0 < 0) return 0;
    const last: number = this.plan.monsterSlots.length - 1;
    if (idx0 > last) return last;
    return idx0;
  }

  private kindForSlot(slotKind: MonsterKind): QuestionKind {
    if (slotKind === MonsterKind.Spelling) {
      return QuestionKind.FillLetter;
    }
    if (slotKind === MonsterKind.Elite || slotKind === MonsterKind.Boss) {
      return this.rng() < 0.5 ? QuestionKind.FillLetter : QuestionKind.Choice;
    }
    return QuestionKind.Choice;
  }

  private pickWordId(slotKind: MonsterKind, lastWordId?: string): string {
    if (slotKind === MonsterKind.Boss && this.bossWordIdsLocal.length > 0) {
      return this.bossWordIdsLocal.shift() as string;
    }
    while (this.wordCursor < this.plan.wordPlan.length) {
      const slot: WordSlot = this.plan.wordPlan[this.wordCursor];
      this.wordCursor += 1;
      if (lastWordId !== undefined && slot.wordId === lastWordId && this.wordCursor < this.plan.wordPlan.length) {
        continue;
      }
      return slot.wordId;
    }
    // wordPlan exhausted — pick any wordId from repo as a final fallback.
    const entries: WordEntry[] = this.repo.all();
    return entries.length > 0 ? entries[0].id : '';
  }
}
```

```typescript
export function formatDayKeyLocal(nowMs: number): string {

  const m: number = d.getMonth() + 1;

  const mm: string = m < 10 ? `0${m}` : `${m}`;
```

```typescript
import preferences from '@ohos.data.preferences';
import { BusinessError } from '@ohos.base';
import { common } from '@kit.AbilityKit';
import {
  CoinSnapshot,
  CoinTxn,
  COIN_PREFS_NAME,
  COIN_PREFS_KEY,
  DAILY_CAP,
  MAX_TXN_HISTORY,
  formatDayKeyLocal,
} from '../models/CoinSnapshot';
import { StringPreferencesLike } from './WrongAnswerStore';

interface CoinSnapshotJson {
  version?: number;
  totalCoins?: number;
  spentCoins?: number;
  todayDayKey?: string;
  todayCoinsEarned?: number;
  todayAdventureCompleted?: boolean;
  txns?: CoinTxnJson[];
}

interface CoinTxnJson {
  ts?: number;
  delta?: number;
  reason?: string;
  balanceAfter?: number;
}

class RealPrefs implements StringPreferencesLike {
  private readonly handle: preferences.Preferences;
  constructor(handle: preferences.Preferences) {
    this.handle = handle;
  }
  async getString(key: string): Promise<string> {
    const v: preferences.ValueType = await this.handle.get(key, '');
    return typeof v === 'string' ? v : '';
  }
  putString(key: string, value: string): Promise<void> {
    return this.handle.put(key, value);
  }
  flush(): Promise<void> {
    return this.handle.flush();
  }
}

export class CoinAccount {
  private prefs?: StringPreferencesLike;
  private snap: CoinSnapshot = new CoinSnapshot();
  private ready: boolean = false;

  injectPreferencesForTest(p: StringPreferencesLike): void {
    this.prefs = p;
    this.ready = true;
  }

  async init(ctx: common.UIAbilityContext): Promise<void> {
    if (this.ready && this.prefs !== undefined) {
      return;
    }
    const real: preferences.Preferences = await preferences.getPreferences(ctx, COIN_PREFS_NAME);
    this.prefs = new RealPrefs(real);
    this.ready = true;
    const raw: string = await this.prefs.getString(COIN_PREFS_KEY);
    if (raw.length > 0) {
      this.snap = parseCoinSnapshot(raw);
    }
  }

  beginToday(nowMs: number): void {
    const key: string = formatDayKeyLocal(nowMs);
    if (this.snap.todayDayKey !== key) {
      this.snap.todayDayKey = key;
      this.snap.todayCoinsEarned = 0;
      this.snap.todayAdventureCompleted = false;
    }
  }

  earn(reason: string, amount: number, nowMs: number): number {
    if (amount <= 0) {
      if (reason === 'today-first') {
        this.snap.todayAdventureCompleted = true;
      }
      return 0;
    }
    const remaining: number = DAILY_CAP - this.snap.todayCoinsEarned;
    const actual: number = Math.max(0, Math.min(amount, remaining));
    if (actual > 0) {
      this.snap.totalCoins += actual;
      this.snap.todayCoinsEarned += actual;
      this.pushTxn(nowMs, actual, reason, this.balance());
    }
    if (reason === 'today-first') {
      this.snap.todayAdventureCompleted = true;
    }
    this.scheduleSave();
    return actual;
  }

  redeem(wishId: string, cost: number): boolean {
    if (cost <= 0) return false;
    if (this.balance() < cost) return false;
    this.snap.spentCoins += cost;
    this.pushTxn(Date.now(), -cost, `redeem:${wishId}`, this.balance());
    this.scheduleSave();
    return true;
  }

  snapshot(): CoinSnapshot {
    const out: CoinSnapshot = new CoinSnapshot();
    out.version = this.snap.version;
    out.totalCoins = this.snap.totalCoins;
    out.spentCoins = this.snap.spentCoins;
    out.todayDayKey = this.snap.todayDayKey;
    out.todayCoinsEarned = this.snap.todayCoinsEarned;
    out.todayAdventureCompleted = this.snap.todayAdventureCompleted;
    out.txns = this.snap.txns.slice();
    return out;
  }

  async flushNow(): Promise<void> {
    if (!this.ready || this.prefs === undefined) return;
    const payload: string = JSON.stringify(this.snap);
    await this.prefs.putString(COIN_PREFS_KEY, payload);
    await this.prefs.flush();
  }

  private balance(): number {
    return this.snap.totalCoins - this.snap.spentCoins;
  }

  private pushTxn(ts: number, delta: number, reason: string, balanceAfter: number): void {
    const t: CoinTxn = new CoinTxn();
    t.ts = ts;
    t.delta = delta;
    t.reason = reason;
    t.balanceAfter = balanceAfter;
    this.snap.txns.push(t);
    while (this.snap.txns.length > MAX_TXN_HISTORY) {
      this.snap.txns.shift();
    }
  }

  private scheduleSave(): void {
    if (!this.ready || this.prefs === undefined) return;
    this.prefs.putString(COIN_PREFS_KEY, JSON.stringify(this.snap)).catch((err: BusinessError): void => {
      console.error(`CoinAccount save failed: ${JSON.stringify(err)}`);
    });
  }
}

export function parseCoinSnapshot(raw: string): CoinSnapshot {
  const out: CoinSnapshot = new CoinSnapshot();
  if (raw.length === 0) return out;
  let parsed: CoinSnapshotJson;
  try {
    parsed = JSON.parse(raw) as CoinSnapshotJson;
  } catch (err) {
    console.error(`parseCoinSnapshot failed: ${JSON.stringify(err as BusinessError)}`);
    return out;
  }
  if (typeof parsed.version === 'number') out.version = parsed.version;
  if (typeof parsed.totalCoins === 'number') out.totalCoins = parsed.totalCoins;
  if (typeof parsed.spentCoins === 'number') out.spentCoins = parsed.spentCoins;
  if (typeof parsed.todayDayKey === 'string') out.todayDayKey = parsed.todayDayKey;
  if (typeof parsed.todayCoinsEarned === 'number') out.todayCoinsEarned = parsed.todayCoinsEarned;
  if (typeof parsed.todayAdventureCompleted === 'boolean') out.todayAdventureCompleted = parsed.todayAdventureCompleted;
  const txns: CoinTxnJson[] | undefined = parsed.txns;
  if (txns !== undefined && txns !== null) {
    for (let i = 0; i < txns.length; i++) {
      const j: CoinTxnJson = txns[i];
      const t: CoinTxn = new CoinTxn();
      t.ts = typeof j.ts === 'number' ? j.ts : 0;
      t.delta = typeof j.delta === 'number' ? j.delta : 0;
      t.reason = typeof j.reason === 'string' ? j.reason : '';
      t.balanceAfter = typeof j.balanceAfter === 'number' ? j.balanceAfter : 0;
      out.txns.push(t);
    }
  }
  return out;
}

  return global;
}

}
```

```typescript
      for (let i = 0; i < this.snap.wishes.length; i++) {
        seen.add(this.snap.wishes[i].id);
      }
      const additions: MagicWish[] = getWishCatalogClone();
      for (let i = 0; i < additions.length; i++) {
        if (!seen.has(additions[i].id)) {
          this.snap.wishes.push(additions[i]);
        }
      }

    for (let i = 0; i < this.snap.wishes.length; i++) {
      const src: MagicWish = this.snap.wishes[i];

      c.id = src.id;
      c.displayName = src.displayName;
      c.costCoins = src.costCoins;
      c.iconEmoji = src.iconEmoji;
      c.state = src.state;
      c.requestedAt = src.requestedAt;
      c.confirmedAt = src.confirmedAt;
      out.push(c);
    }
    return out;

  request(wishId: string, account: CoinAccount, nowMs: number): boolean {
    const w: MagicWish | undefined = this.findWish(wishId);
    if (w === undefined) return false;
    if (w.state !== WishState.Idle) return false;
    const balance: number = account.snapshot().totalCoins - account.snapshot().spentCoins;
    if (balance < w.costCoins) return false;
    w.state = WishState.Pending;
    w.requestedAt = nowMs;
    this.scheduleSave();
    return true;
  }

  confirmByParent(wishId: string, account: CoinAccount, nowMs: number): boolean {
    const w: MagicWish | undefined = this.findWish(wishId);
    if (w === undefined) return false;
    if (w.state !== WishState.Pending) return false;
    const ok: boolean = account.redeem(w.id, w.costCoins);
    if (!ok) return false;
    w.state = WishState.Confirmed;
    w.confirmedAt = nowMs;
    this.scheduleSave();
    return true;
  }

  cancel(wishId: string): void {
    const w: MagicWish | undefined = this.findWish(wishId);
    if (w === undefined) return;
    if (w.state === WishState.Pending) {
      w.state = WishState.Idle;
      w.requestedAt = 0;
      this.scheduleSave();
    }
  }

  acknowledge(wishId: string): void {
    const w: MagicWish | undefined = this.findWish(wishId);
    if (w === undefined) return;
    if (w.state === WishState.Confirmed) {
      w.state = WishState.Idle;
      w.requestedAt = 0;
      w.confirmedAt = 0;
      this.scheduleSave();
    }
  }

  async flushNow(): Promise<void> {
    if (!this.ready || this.prefs === undefined) return;
    await this.prefs.putString(WISHLIST_PREFS_KEY, JSON.stringify(this.snap));
    await this.prefs.flush();
  }

  private findWish(wishId: string): MagicWish | undefined {
    for (let i = 0; i < this.snap.wishes.length; i++) {
      if (this.snap.wishes[i].id === wishId) {
        return this.snap.wishes[i];
      }
    }
    return undefined;
  }

      console.error(`WishlistStore save failed: ${JSON.stringify(err)}`);
    });
  }

export function parseWishlistSnapshot(raw: string): WishlistSnapshot {
  const out: WishlistSnapshot = new WishlistSnapshot();
  if (raw.length === 0) return out;
  let parsed: WishlistSnapshotJson;
  try {
    parsed = JSON.parse(raw) as WishlistSnapshotJson;
  } catch (err) {
    console.error(`parseWishlistSnapshot failed: ${JSON.stringify(err as BusinessError)}`);
    return out;
  }
  if (typeof parsed.version === 'number') out.version = parsed.version;
  const xs: MagicWishJson[] | undefined = parsed.wishes;
  if (xs !== undefined && xs !== null) {
    for (let i = 0; i < xs.length; i++) {
      const j: MagicWishJson = xs[i];
      const w: MagicWish = new MagicWish();
      w.id = typeof j.id === 'string' ? j.id : '';
      w.displayName = typeof j.displayName === 'string' ? j.displayName : '';
      w.costCoins = typeof j.costCoins === 'number' ? j.costCoins : 0;
      w.iconEmoji = typeof j.iconEmoji === 'string' ? j.iconEmoji : '';
      w.state = typeof j.state === 'string' ? (j.state as WishState) : WishState.Idle;
      w.requestedAt = typeof j.requestedAt === 'number' ? j.requestedAt : 0;
      w.confirmedAt = typeof j.confirmedAt === 'number' ? j.confirmedAt : 0;
      if (w.id.length > 0) {
        out.wishes.push(w);
      }
    }
  }
  return out;
}
```

```typescript
function computeReviewAccuracy(plan: TodaySessionPlan, answers: Map<string, AnswerCount>): number {
  let correct: number = 0;
  let total: number = 0;
  for (let i = 0; i < plan.wordPlan.length; i++) {
    const slot: WordSlot = plan.wordPlan[i];
    if (slot.source !== WordSource.Review) continue;
    const cnt: AnswerCount | undefined = answers.get(slot.wordId);
    if (cnt === undefined) continue;
    correct += cnt.correct;
```

```typescript
const HOLD_DURATION_MS: number = 3000;
const TICK_MS: number = 33;
const PROGRESS_PER_TICK: number = TICK_MS / HOLD_DURATION_MS;

/**
 * ParentLongPressGate — circular progress button that fires
 * onConfirm only after a 3-second hold. Releasing early resets
 * the progress so accidental taps never confirm.
 */
@Component
export struct ParentLongPressGate {
  @Prop label: string = '长按 3 秒由家长确认';
  onConfirm: () => void = (): void => {};

  @State progress: number = 0;
  @State holding: boolean = false;
  private timer: number = -1;

  build(): void {
    Stack({ alignContent: Alignment.Center }) {
      Circle({ width: 80, height: 80 })
        .fill('#FCEAEA');
      Circle({ width: 80, height: 80 })
        .stroke('#E63946')
        .strokeWidth(6)
        .fillOpacity(0)
        .strokeDashArray([Math.PI * 80 * this.progress, Math.PI * 80])
        .strokeLineCap(LineCapStyle.Round);
      Text(this.label)
        .id('ParentLongPressGateLabel')
        .fontSize(11)
        .fontColor('#E63946')
        .textAlign(TextAlign.Center)
        .padding(8);
    }
    .id('ParentLongPressGate')
    .width(120)
    .height(120)
    .onTouch((event: TouchEvent): void => {
      if (event.type === TouchType.Down) {
        this.startHold();
      } else if (event.type === TouchType.Up || event.type === TouchType.Cancel) {
        this.endHold();
      }
    });
  }

  private startHold(): void {
    if (this.timer >= 0) return;
    this.holding = true;
    this.timer = setInterval((): void => {
      this.progress = Math.min(1, this.progress + PROGRESS_PER_TICK);
      if (this.progress >= 1) {
        this.fireConfirm();
      }
    }, TICK_MS);
  }

  private endHold(): void {
    this.clearTimer();
    if (this.progress < 1) {
      this.progress = 0;
    }
    this.holding = false;
  }

  private fireConfirm(): void {
    this.clearTimer();
    this.holding = false;
    const cb: () => void = this.onConfirm;
    setTimeout((): void => {
      cb();
      this.progress = 0;
    }, 200);
  }

  private clearTimer(): void {
    if (this.timer >= 0) {
      clearInterval(this.timer);
      this.timer = -1;
    }
  }

  aboutToDisappear(): void {
    this.clearTimer();
  }
}
```
