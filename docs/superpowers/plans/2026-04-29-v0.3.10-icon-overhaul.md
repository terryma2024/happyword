### Task 2: Add `tools/recraft/svg-to-png.mjs`

**Files:** Create `tools/recraft/svg-to-png.mjs`.

- [ ] **Step 2.1: Implement minimal CLI**

```javascript
#!/usr/bin/env node
// Wraps `rsvg-convert -w <px> -h <px>` to rasterise a Recraft-generated
// SVG into a PNG at a given pixel size. Used by icons-to-launcher.sh
// to produce 1024×1024 / 216×216 launcher artefacts from the generated
// SVGs without depending on a Node SVG library.

import { mkdir } from 'node:fs/promises';
import { dirname } from 'node:path';
import { spawnSync } from 'node:child_process';

function parseArgs(argv) {
  const args = {};
  for (let i = 0; i < argv.length; i += 1) {
    const a = argv[i];
    if (!a.startsWith('--')) throw new Error(`Unknown arg: ${a}`);
    const key = a.slice(2);
    const next = argv[i + 1];
    if (!next || next.startsWith('--')) throw new Error(`Missing value for ${a}`);
    args[key] = next;
    i += 1;
  }
  return args;
}

async function main() {
  const args = parseArgs(process.argv.slice(2));
  const inFile = args.in;
  const outFile = args.out;
  const size = Number.parseInt(args.size, 10);
  if (!inFile || !outFile || !Number.isInteger(size) || size <= 0) {
    throw new Error('Usage: svg-to-png.mjs --in <svg> --out <png> --size <px>');
  }
  await mkdir(dirname(outFile), { recursive: true });
  const r = spawnSync('rsvg-convert', [
    '-w', String(size),
    '-h', String(size),
    '-o', outFile,
    inFile,
  ], { stdio: 'inherit' });
  if (r.status !== 0) {
    throw new Error(`rsvg-convert exited with code ${r.status}`);
  }
  console.log(`[ok] ${inFile} → ${outFile} (${size}×${size})`);
}

main().catch((err) => {
  console.error(err.message);
  process.exit(1);
});
```
