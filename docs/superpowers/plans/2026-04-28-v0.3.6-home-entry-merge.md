# V0.3.6 Home Entry Merge Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Collapse HomePage start/today CTAs into a single `HomeStartButton` driving today's adventure, relocate the review entry to the top toolbar with a locked-state inline toast, and drop the dev-only `TtsTestPage`.

**Architecture:** Single-page UI surgery on `HomePage.ets` (plus one route-list and one independent diagnostic page deletion). 3 isolated commits, each independently buildable and testable, ordered from lowest-risk (TTS purge) to highest-risk (toast TDD addition). Reuses existing UI test helpers (`clickByIdShared`, `launchAppShared`) from `RoutingFlow.ui.test.ets`. Validates via existing 6 UI suites as regression layer plus 1 new locked-toast suite.

**Tech Stack:** ArkTS / HarmonyOS NEXT; `@kit.TestKit` for UiTest; Hypium for assertions; @Builder inline composition (no @Component creation per project codelinter rule); ArkUI `setTimeout` / `clearTimeout` (global) for toast self-hide.

```typescript
import { describe, it, expect } from '@ohos/hypium';
import { abilityDelegatorRegistry, Driver, ON, Component } from '@kit.TestKit';
import { Want } from '@kit.AbilityKit';
import { clickByIdShared } from './RoutingFlow.ui.test';

const BUNDLE: string = 'com.terryma.wordmagicgame';
const DELEGATOR: abilityDelegatorRegistry.AbilityDelegator =
  abilityDelegatorRegistry.getAbilityDelegator();

async function launchApp(): Promise<Driver> {
  const want: Want = { bundleName: BUNDLE, abilityName: 'EntryAbility' };
  await DELEGATOR.startAbility(want);
  const driver: Driver = Driver.create();
  await driver.delayMs(1000);
  return driver;
}

/**
 * v0.3.6 toolbar acceptance:
 *  - 📚 HomeReviewButton lives in the top toolbar (not the deleted
 *    right column), and the dev-only 🧪 HomeTtsTestButton is gone.
 *  - On a fresh cold launch (0 wrong words → review locked), tapping
 *    the review button surfaces an inline toast (HomeReviewLockedToast)
 *    that auto-dismisses inside 2s without navigating away.
 */
export default function homeToolbarLockedUiTest(): void {
  describe('HomeToolbarLocked', () => {
    it('topToolbarHasReviewNotTts', 0, async (done: Function) => {
      const driver: Driver = await launchApp();
      await driver.assertComponentExist(ON.id('HomeReviewButton'));
      const tts: Component | null =
        await driver.findComponent(ON.id('HomeTtsTestButton'));
      expect(tts === null).assertTrue();
      done();
    });

    it('lockedReviewShowsToastAndStays', 0, async (done: Function) => {
      const driver: Driver = await launchApp();
      await clickByIdShared(driver, 'HomeReviewButton');
      await driver.delayMs(300);
      await driver.assertComponentExist(ON.id('HomeReviewLockedToast'));
      await driver.assertComponentExist(ON.id('HomeStartButton'));
      await driver.delayMs(1800);
      const toast: Component | null =
        await driver.findComponent(ON.id('HomeReviewLockedToast'));
      expect(toast === null).assertTrue();
      done();
    });
  });
}
```
