# V0.5.1 Walking Skeleton Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** End-to-end walking skeleton — Vercel-deployed FastAPI server with JWT login, MongoDB seed data, and a public `/packs/latest.json` endpoint; HarmonyOS client cold-starts via cache → rawfile fallback → async refresh.

**Architecture:** Monorepo `server/` subdirectory holds a FastAPI + uv + Beanie ODM Python project deployed via Vercel python runtime. Client adds three new services (`WordPackCache`, `RemoteWordPackService`, `WordPackBootstrapper`) that wrap existing `WordRepository` for safe cold-start with rawfile fallback. No client UI changes; existing UI/unit tests must not regress.

**Tech Stack:** Python 3.12, FastAPI, uv, Beanie + Motor (MongoDB Atlas), python-jose + passlib (JWT + bcrypt), pydantic-settings, httpx, mongomock-motor, pytest. Client uses ArkTS + `@kit.NetworkKit` HTTP, `@ohos.data.preferences`, `@ohos/hypium` tests.

**Spec:** [docs/superpowers/specs/2026-04-30-v0.5-content-backend-design.md §5.1](../specs/2026-04-30-v0.5-content-backend-design.md)

```python
import time
from contextlib import asynccontextmanager
from typing import AsyncIterator

from beanie import init_beanie
from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware
from motor.motor_asyncio import AsyncIOMotorClient

from app.config import get_settings
from app.models.user import User
from app.models.word import Word


@asynccontextmanager
async def lifespan(app: FastAPI) -> AsyncIterator[None]:
    settings = get_settings()
    client = AsyncIOMotorClient(settings.mongo_uri)
    await init_beanie(
        database=client[settings.mongo_db_name],
        document_models=[User, Word],
    )
    app.state.mongo_client = client
    try:
        yield
    finally:
        client.close()


app = FastAPI(title="happyword-server", version="0.5.1", lifespan=lifespan)

app.add_middleware(
    CORSMiddleware,
    allow_origins=[get_settings().cors_allow_origins],
    allow_methods=["*"],
    allow_headers=["*"],
)


@app.get("/api/v1/health")
async def health() -> dict[str, object]:
    return {"ok": True, "ts": int(time.time())}
```

```python
from passlib.context import CryptContext

_pwd_ctx = CryptContext(schemes=["bcrypt"], deprecated="auto")


def hash_password(plain: str) -> str:
    return _pwd_ctx.hash(plain)


def verify_password(plain: str, hashed: str) -> bool:
    return _pwd_ctx.verify(plain, hashed)
```

```python
_pwd_ctx = CryptContext(schemes=["bcrypt"], deprecated="auto")


class JwtError(Exception):
    """Raised when a token cannot be verified."""


def hash_password(plain: str) -> str:
    return _pwd_ctx.hash(plain)


def verify_password(plain: str, hashed: str) -> bool:
    return _pwd_ctx.verify(plain, hashed)


def create_access_token(subject: str, expires_in: int | None = None) -> str:
    settings = get_settings()
    if expires_in is None:
        expires_in = settings.jwt_expire_hours * 3600
    now = int(time.time())
    payload = {"sub": subject, "iat": now, "exp": now + expires_in}
    return jwt.encode(payload, settings.jwt_secret, algorithm="HS256")


def decode_access_token(token: str) -> dict[str, Any]:
    settings = get_settings()
    try:
        return jwt.decode(token, settings.jwt_secret, algorithms=["HS256"])
    except JWTError as e:
        raise JwtError(str(e)) from e
```

```python
from fastapi import Depends, HTTPException, status
from fastapi.security import HTTPAuthorizationCredentials, HTTPBearer

from app.models.user import User
from app.services.auth_service import JwtError, decode_access_token

_bearer = HTTPBearer(auto_error=False)


async def current_user(
    creds: HTTPAuthorizationCredentials | None = Depends(_bearer),
) -> User:
    if creds is None or creds.scheme.lower() != "bearer":
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail={"error": {"code": "UNAUTHORIZED", "message": "Missing bearer token"}},
        )
    try:
        payload = decode_access_token(creds.credentials)
    except JwtError:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail={"error": {"code": "UNAUTHORIZED", "message": "Invalid or expired token"}},
        ) from None
    sub = payload.get("sub")
    if not isinstance(sub, str):
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail={"error": {"code": "UNAUTHORIZED", "message": "Token missing subject"}},
        )
    user = await User.find_one(User.username == sub)
    if user is None:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail={"error": {"code": "UNAUTHORIZED", "message": "User not found"}},
        )
    return user
```

```python
"""One-shot CLI to create or reset an admin user. Idempotent on username.

Usage:
    cd server
    uv run python scripts/create_admin_user.py <username> <password>

If the user exists, the password is RESET (this is intentional — it's a recovery tool).
"""
import asyncio
import sys
from datetime import datetime, timezone

from beanie import init_beanie
from motor.motor_asyncio import AsyncIOMotorClient

from app.config import get_settings
from app.models.user import User, UserRole
from app.services.auth_service import hash_password


async def main(username: str, password: str) -> None:
    settings = get_settings()
    client = AsyncIOMotorClient(settings.mongo_uri)
    try:
        await init_beanie(database=client[settings.mongo_db_name], document_models=[User])
        existing = await User.find_one(User.username == username)
        if existing is None:
            await User(
                username=username,
                password_hash=hash_password(password),
                role=UserRole.ADMIN,
                created_at=datetime.now(tz=timezone.utc),
            ).insert()
            print(f"Created admin user '{username}'.")
        else:
            existing.password_hash = hash_password(password)
            await existing.save()
            print(f"Reset password for existing admin user '{username}'.")
    finally:
        client.close()


if __name__ == "__main__":
    if len(sys.argv) != 3:
        sys.stderr.write("Usage: create_admin_user.py <username> <password>\n")
        sys.exit(2)
```

```typescript
import preferences from '@ohos.data.preferences';
import { common } from '@kit.AbilityKit';
import { BusinessError } from '@ohos.base';
import { StringPreferencesLike } from './WrongAnswerStore';

const PREFS_NAME: string = 'word_pack_cache';
const PACK_KEY: string = 'pack_v1';

class RealPrefs implements StringPreferencesLike {
  private readonly handle: preferences.Preferences;
  constructor(handle: preferences.Preferences) {
    this.handle = handle;
  }
  async getString(key: string): Promise<string> {
    const v: preferences.ValueType = await this.handle.get(key, '');
    return typeof v === 'string' ? v : '';
  }
  putString(key: string, value: string): Promise<void> {
    return this.handle.put(key, value);
  }
  flush(): Promise<void> {
    return this.handle.flush();
  }
}

/**
 * V0.5.1: persistent cache for the remote word pack JSON.
 *
 * Stored as a single string (the entire pack JSON envelope). Parsing happens
 * in WordPackBootstrapper / WordRepository. This class only enforces:
 *   - non-empty string round-trip
 *   - reject obviously corrupt payloads (parse fails, missing `words` array)
 */
export class WordPackCache {
  private prefs?: StringPreferencesLike;

  injectPreferencesForTest(p: StringPreferencesLike): void {
    this.prefs = p;
  }

  async init(ctx: common.UIAbilityContext): Promise<void> {
    if (this.prefs !== undefined) {
      return;
    }
    try {
      const handle: preferences.Preferences = await preferences.getPreferences(ctx, PREFS_NAME);
      this.prefs = new RealPrefs(handle);
    } catch (err) {
      const be: BusinessError = err as BusinessError;
      console.error(`WordPackCache.init failed: ${JSON.stringify(be)}`);
    }
  }

  async read(): Promise<string> {
    if (this.prefs === undefined) {
      return '';
    }
    const raw: string = await this.prefs.getString(PACK_KEY);
    if (raw.length === 0) {
      return '';
    }
    if (!WordPackCache.looksValid(raw)) {
      return '';
    }
    return raw;
  }

  async write(packJson: string): Promise<void> {
    if (this.prefs === undefined) {
      return;
    }
    if (!WordPackCache.looksValid(packJson)) {
      return;
    }
    await this.prefs.putString(PACK_KEY, packJson);
    await this.prefs.flush();
  }

  static looksValid(raw: string): boolean {
    if (raw.length === 0) {
      return false;
    }
    let parsed: object;
    try {
      parsed = JSON.parse(raw) as object;
    } catch (_err) {
      return false;
    }
    if (parsed === null || parsed === undefined) {
      return false;
    }
    const obj: Record<string, Object> = parsed as Record<string, Object>;
    const words: Object | undefined = obj.words;
    if (words === undefined || words === null) {
      return false;
    }
    return Array.isArray(words);
  }
}
```

```typescript
import { http } from '@kit.NetworkKit';
import { BusinessError } from '@ohos.base';
import { REMOTE_FETCH_TIMEOUT_MS } from './RemoteWordPackConfig';

export class RemoteFetchResult {
  status: number = 0;       // 0 means "did not complete (network error / timeout / disabled)"
  body: string = '';
  etag: string = '';
}

export interface HttpRequester {
  fetch(url: string, ifNoneMatch: string | undefined, timeoutMs: number): Promise<RemoteFetchResult>;
}

class RealHttpRequester implements HttpRequester {
  async fetch(url: string, ifNoneMatch: string | undefined, timeoutMs: number): Promise<RemoteFetchResult> {
    const req: http.HttpRequest = http.createHttp();
    const headers: Record<string, string> = { 'Accept': 'application/json' };
    if (ifNoneMatch !== undefined && ifNoneMatch.length > 0) {
      headers['If-None-Match'] = ifNoneMatch;
    }
    try {
      const resp: http.HttpResponse = await req.request(url, {
        method: http.RequestMethod.GET,
        header: headers,
        connectTimeout: timeoutMs,
        readTimeout: timeoutMs,
      });
      const out: RemoteFetchResult = new RemoteFetchResult();
      out.status = resp.responseCode;
      out.body = typeof resp.result === 'string' ? resp.result : JSON.stringify(resp.result);
      const respHeader: Record<string, string> = (resp.header ?? {}) as Record<string, string>;
      out.etag = respHeader['etag'] ?? respHeader['ETag'] ?? '';
      return out;
    } finally {
      req.destroy();
    }
  }
}

/**
 * V0.5.1: thin HTTP wrapper around the public /packs/latest.json endpoint.
 *
 * Always returns a RemoteFetchResult, never throws — callers just check `status`.
 * Status 0 = "cannot use", 200 = fresh body, 304 = cache still valid.
 */
export class RemoteWordPackService {
  private requester: HttpRequester = new RealHttpRequester();

  injectHttpRequesterForTest(r: HttpRequester): void {
    this.requester = r;
  }

  async fetchLatest(url: string, ifNoneMatch: string | undefined): Promise<RemoteFetchResult> {
    if (url.length === 0) {
      return new RemoteFetchResult();
    }
    try {
      return await this.requester.fetch(url, ifNoneMatch, REMOTE_FETCH_TIMEOUT_MS);
    } catch (err) {
      const be: BusinessError = err as BusinessError;
      console.error(`RemoteWordPackService.fetchLatest failed: ${JSON.stringify(be)}`);
      return new RemoteFetchResult();
    }
  }
}
```

```typescript
import { common } from '@kit.AbilityKit';
import { WordEntry } from '../models/WordEntry';
import { WordRepository } from './WordRepository';
import { WordPackCache } from './WordPackCache';
import { RemoteWordPackService, RemoteFetchResult } from './RemoteWordPackService';
import { latestPackUrl } from './RemoteWordPackConfig';

export type RawfileLoader = () => Promise<WordEntry[]>;

export interface BootstrapDeps {
  cache: WordPackCache;
  remote: RemoteWordPackService;
  rawfileLoader: RawfileLoader;
  serverUrl: string;
}

/**
 * V0.5.1: cold-start orchestrator for the word pack.
 *
 * Sync section (returns within tens of ms):
 *   1. read cache
 *   2. if valid -> WordRepository from cache JSON
 *   3. else -> WordRepository from rawfile loader
 *
 * Async section (fire-and-forget, no UI block):
 *   4. fetch latest pack with If-None-Match
 *   5. on 200 -> write cache (next cold start picks it up)
 *   6. on 304 / error -> keep cache as-is
 *
 * Critical invariant: the in-memory `WordRepository` returned from `bootstrap`
 * is NOT replaced if a remote refresh succeeds. The new pack only takes effect
 * on the next cold start, preventing word-pool swaps mid-battle.
 */
export class WordPackBootstrapper {
  /**
   * Production helper: resolve real cache + service from the ability context.
   */
  static async forContext(ctx: common.UIAbilityContext): Promise<BootstrapDeps> {
    const cache: WordPackCache = new WordPackCache();
    await cache.init(ctx);
    const remote: RemoteWordPackService = new RemoteWordPackService();
    const repo: WordRepository = new WordRepository();
    const deps: BootstrapDeps = {
      cache,
      remote,
      rawfileLoader: async (): Promise<WordEntry[]> => {
        await repo.loadFromRawfile(ctx);
        return repo.all();
      },
      serverUrl: latestPackUrl(),
    };
    return deps;
  }

  async bootstrap(deps: BootstrapDeps): Promise<WordRepository> {
    const cached: string = await deps.cache.read();
    const repo: WordRepository = new WordRepository();

    if (cached.length > 0) {
      const entries: WordEntry[] = WordRepository.parseWordsJson(cached);
      if (entries.length > 0) {
        repo.setEntries(entries);
      } else {
        repo.setEntries(await deps.rawfileLoader());
      }
    } else {
      repo.setEntries(await deps.rawfileLoader());
    }

    this.refreshInBackground(deps);
    return repo;
  }

  private refreshInBackground(deps: BootstrapDeps): void {
    void (async (): Promise<void> => {
      try {
        const res: RemoteFetchResult = await deps.remote.fetchLatest(deps.serverUrl, undefined);
        if (res.status !== 200 || res.body.length === 0) {
          return;
        }
        await deps.cache.write(res.body);
      } catch (err) {
        console.error(`WordPackBootstrapper.refresh failed: ${JSON.stringify(err)}`);
      }
    })();
  }
}
```
