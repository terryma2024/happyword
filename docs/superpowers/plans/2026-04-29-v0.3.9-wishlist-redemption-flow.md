```typescript
/**
 * One redemption recorded after the parent PIN gate clears and
 * CoinAccount.redeem succeeds. Stored plaintext in preferences;
 * RedemptionHistoryStore caps the persisted list at
 * REDEMPTION_HISTORY_CAP entries (oldest-shifted FIFO).
 *
 * Schema is intentionally minimal — V0.3.9 spec §5/§8.1 chose
 * "time + emoji+name + cost" as the smallest set that gives the
 * parent enough context to reconcile what was redeemed.
 */
export class RedemptionRecord {
  /** Unique id: `${ts}-${random4hex}`; collisions improbable in single-device scope. */
  id: string = '';
  /** Date.now() at the moment markConfirmed succeeds. */
  ts: number = 0;
  wishId: string = '';
  displayName: string = '';
  iconEmoji: string = '';
  costCoins: number = 0;
}

export class RedemptionHistorySnapshot {
  version: number = 1;
  records: RedemptionRecord[] = [];
}

export const REDEMPTION_PREFS_NAME: string = 'wordmagic_redemption_history';
export const REDEMPTION_PREFS_KEY: string = 'snapshot_v1';
export const REDEMPTION_HISTORY_CAP: number = 50;

/**
 * Format a Date.now() millisecond value as `YYYY-MM-DD HH:mm` in
 * the device's local timezone. Two-digit zero-padding throughout.
 * Pure — used by RedemptionHistoryPage row rendering.
 */
export function formatLocalTimestamp(ms: number): string {
  const d: Date = new Date(ms);
  const yyyy: string = `${d.getFullYear()}`;
  const mm: string = pad2(d.getMonth() + 1);
  const dd: string = pad2(d.getDate());
  const hh: string = pad2(d.getHours());
  const mi: string = pad2(d.getMinutes());
  return `${yyyy}-${mm}-${dd} ${hh}:${mi}`;
}

function pad2(n: number): string {
  return n < 10 ? `0${n}` : `${n}`;
}
```

```typescript
import preferences from '@ohos.data.preferences';
import { BusinessError } from '@ohos.base';
import { common } from '@kit.AbilityKit';
import {
  RedemptionRecord,
  RedemptionHistorySnapshot,
  REDEMPTION_PREFS_NAME,
  REDEMPTION_PREFS_KEY,
  REDEMPTION_HISTORY_CAP,
} from '../models/RedemptionRecord';
import { StringPreferencesLike } from './WrongAnswerStore';

interface RedemptionRecordJson {
  id?: string;
  ts?: number;
  wishId?: string;
  displayName?: string;
  iconEmoji?: string;
  costCoins?: number;
}

interface RedemptionHistorySnapshotJson {
  version?: number;
  records?: RedemptionRecordJson[];
}

class RealPrefs implements StringPreferencesLike {
  private readonly handle: preferences.Preferences;
  constructor(handle: preferences.Preferences) {
    this.handle = handle;
  }
  async getString(key: string): Promise<string> {
    const v: preferences.ValueType = await this.handle.get(key, '');
    return typeof v === 'string' ? v : '';
  }
  putString(key: string, value: string): Promise<void> {
    return this.handle.put(key, value);
  }
  flush(): Promise<void> {
    return this.handle.flush();
  }
}

/**
 * RedemptionHistoryStore mirrors WishlistStore's preferences pattern.
 * In-memory mutations are synchronous; persistence is fire-and-forget
 * via putString. Tests inject a Map-backed prefs stand-in via
 * injectPreferencesForTest. Cap enforced on every add().
 */
export class RedemptionHistoryStore {
  private prefs?: StringPreferencesLike;
  private snap: RedemptionHistorySnapshot = new RedemptionHistorySnapshot();
  private ready: boolean = false;

  injectPreferencesForTest(p: StringPreferencesLike): void {
    this.prefs = p;
    this.ready = true;
  }

  async init(ctx: common.UIAbilityContext): Promise<void> {
    if (this.ready && this.prefs !== undefined) {
      return;
    }
    const real: preferences.Preferences =
      await preferences.getPreferences(ctx, REDEMPTION_PREFS_NAME);
    this.prefs = new RealPrefs(real);
    this.ready = true;
    await this.rehydrateForTest();
  }

  /**
   * Internal rehydrate hook exposed for unit tests so they can simulate
   * a cold-start reload without spinning up the full UIAbilityContext.
   */
  async rehydrateForTest(): Promise<void> {
    if (this.prefs === undefined) {
      return;
    }
    const raw: string = await this.prefs.getString(REDEMPTION_PREFS_KEY);
    if (raw.length > 0) {
      this.snap = parseRedemptionHistorySnapshot(raw);
    }
  }

  add(record: RedemptionRecord): void {
    const copy: RedemptionRecord = cloneRecord(record);
    this.snap.records.push(copy);
    while (this.snap.records.length > REDEMPTION_HISTORY_CAP) {
      this.snap.records.shift();
    }
    this.scheduleSave();
  }

  /** Returns deep-cloned records sorted newest first (ts desc). */
  list(): RedemptionRecord[] {
    const out: RedemptionRecord[] = [];
    for (let i = 0; i < this.snap.records.length; i++) {
      out.push(cloneRecord(this.snap.records[i]));
    }
    out.sort((a: RedemptionRecord, b: RedemptionRecord): number => b.ts - a.ts);
    return out;
  }

  async flushNow(): Promise<void> {
    if (!this.ready || this.prefs === undefined) {
      return;
    }
    await this.prefs.putString(
      REDEMPTION_PREFS_KEY, JSON.stringify(this.snap));
    await this.prefs.flush();
  }

  private scheduleSave(): void {
    if (!this.ready || this.prefs === undefined) {
      return;
    }
    this.prefs.putString(REDEMPTION_PREFS_KEY, JSON.stringify(this.snap))
      .catch((err: BusinessError): void => {
        console.error(`RedemptionHistoryStore save failed: ${JSON.stringify(err)}`);
      });
  }
}

function cloneRecord(src: RedemptionRecord): RedemptionRecord {
  const c: RedemptionRecord = new RedemptionRecord();
  c.id = src.id;
  c.ts = src.ts;
  c.wishId = src.wishId;
  c.displayName = src.displayName;
  c.iconEmoji = src.iconEmoji;
  c.costCoins = src.costCoins;
  return c;
}

export function parseRedemptionHistorySnapshot(raw: string): RedemptionHistorySnapshot {
  const out: RedemptionHistorySnapshot = new RedemptionHistorySnapshot();
  if (raw.length === 0) {
    return out;
  }
  let parsed: RedemptionHistorySnapshotJson;
  try {
    parsed = JSON.parse(raw) as RedemptionHistorySnapshotJson;
  } catch (err) {
    console.error(`parseRedemptionHistorySnapshot failed: ${JSON.stringify(err as BusinessError)}`);
    return out;
  }
  if (parsed === null || parsed === undefined) {
    return out;
  }
  if (typeof parsed.version === 'number') {
    out.version = parsed.version;
  }
  const xs: RedemptionRecordJson[] | undefined = parsed.records;
  if (xs !== undefined && xs !== null) {
    for (let i = 0; i < xs.length; i++) {
      const j: RedemptionRecordJson = xs[i];
      const r: RedemptionRecord = new RedemptionRecord();
      r.id = typeof j.id === 'string' ? j.id : '';
      r.ts = typeof j.ts === 'number' ? j.ts : 0;
      r.wishId = typeof j.wishId === 'string' ? j.wishId : '';
      r.displayName = typeof j.displayName === 'string' ? j.displayName : '';
      r.iconEmoji = typeof j.iconEmoji === 'string' ? j.iconEmoji : '';
      r.costCoins = typeof j.costCoins === 'number' ? j.costCoins : 0;
      if (r.id.length > 0) {
        out.records.push(r);
      }
    }
  }
  return out;
}

let globalStore: RedemptionHistoryStore | undefined = undefined;

export function getRedemptionHistoryStore(): RedemptionHistoryStore {
  if (globalStore === undefined) {
    globalStore = new RedemptionHistoryStore();
  }
  return globalStore;
}

export function __resetRedemptionHistoryStoreForTest(): void {
  globalStore = undefined;
}
```

import { UIContext } from '@kit.ArkUI';

export const PIN_LENGTH: number = 6;

/** Pure: returns true iff input.length === PIN_LENGTH and input === stored. */
export function validatePin(input: string, stored: string): boolean {
  if (stored.length !== PIN_LENGTH) {
    return false;
  }
  if (input.length !== PIN_LENGTH) {
    return false;
  }
  return input === stored;
}

/** Pure: append `digit` to `buf` unless buf already has PIN_LENGTH chars. */
export function pushDigit(buf: string, digit: string): string {
  if (buf.length >= PIN_LENGTH) {
    return buf;
  }
  return buf + digit;
}

/** Pure: remove last char of buf; no-op on empty. */
export function popDigit(buf: string): string {
  if (buf.length === 0) {
    return '';
  }
  return buf.substring(0, buf.length - 1);
}

      ui.animateTo({ duration: SHAKE_STEP_MS, curve: Curve.Linear }, (): void => {

  private onApplyTap(wish: MagicWish): void {
    const cfg: GameConfig | undefined =
      AppStorage.get<GameConfig>(GAME_CONFIG_STORAGE_KEY);
    const pin: string = (cfg !== undefined) ? cfg.parentPin : '';
    if (pin.length !== 6) {

    this.pendingWish = wish;
    this.pinDialog = new CustomDialogController({
      builder: ParentPinDialog({
        storedPin: pin,
        onCorrect: (): void => this.onPinCorrect(),
        onCancel: (): void => this.onPinCancel(),
      }),
      autoCancel: true,
      customStyle: false,
    });
    this.pinDialog.open();
  }

  private showPinNotConfiguredHint(): void {
    AlertDialog.show({
      title: '家长密码尚未设置',
      message: '请先在 ⚙ 设置 → 家长密码 中创建 6 位数字密码。',
      confirm: { value: '知道了', action: (): void => {} },
    });

  private onPinCancel(): void {
    this.pendingWish = undefined;
  }

  private onPinCorrect(): void {
    const w: MagicWish | undefined = this.pendingWish;
    this.pendingWish = undefined;
    if (w === undefined) {
      return;
    }
    const ok: boolean = this.coins.redeem(w.id, w.costCoins);
    if (!ok) {
      console.warn(`WishlistPage redeem rejected for ${w.id}`);
      return;
    }
    const record: RedemptionRecord = new RedemptionRecord();
    record.id = `${Date.now()}-${Math.floor(Math.random() * 0xffff).toString(16)}`;
    record.ts = Date.now();
    record.wishId = w.id;
    record.displayName = w.displayName;
    record.iconEmoji = w.iconEmoji;
    record.costCoins = w.costCoins;
    this.history.add(record);
    this.store.markConfirmed(w.id, Date.now());
    this.refresh();
    this.startGiftBoxCycle(w.id);    // Task 12 wires this
  }
