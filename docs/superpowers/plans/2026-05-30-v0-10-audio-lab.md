# V0.10 Audio Lab Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a debug-only HarmonyOS audio lab that lets developers manually compare BGM, SFX, TTS, music lowering, single resume, and SFX-during-voice policies without changing production BattlePage behavior.

**Architecture:** Add an isolated `audio_lab/` module with typed config/events/snapshots, an `AudioLabController`, fakeable lane interfaces, and real HarmonyOS lane implementations. Add the PCM-only `AudioLabPage` under the debug `DevMenuPage` launcher; production battle code remains untouched. Keep backend environment switching on a separate `DomainSwitchPage` so `Domain Switch`, `PcmAudioLab`, and `MessageBubbleLab` are peer debug tools.

**Tech Stack:** HarmonyOS NEXT, ArkTS / ArkUI, `@ohos.multimedia.media` AVPlayer / AudioRenderer, CoreSpeechKit PCM synthesis, Hypium local unit tests, Hvigor build, CodeLinter.

---

## Current Harmony Delta To Preserve For Replication

The implementation evolved after the original plan:

- The final developer entry is `PcmAudioLab`, not generic `Audio Lab`.
- The System TTS backend switch was removed. Manual testing showed System TTS direct playback could be silent and/or interrupt BGM, so the selected route is PCM-only.
- `DevMenuPage` is now a launcher page with three sibling cards: `Domain Switch`, `PcmAudioLab`, `MessageBubbleLab`.
- Backend routing tools moved from `DevMenuPage` into `DomainSwitchPage`.
- `DevMenuToolEntries` owns the debug launcher model and is covered by a no-device unit test.
- Harmony keeps DevMenu/triple-tap out of ohosTest per project policy; do not add flaky UI automation through the hidden debug gesture.

iOS and Android replication plans should start from this final shape rather than from the early single-page DevMenu or System TTS A/B design.

```ts
import { describe, it, expect } from '@ohos/hypium';
import { common } from '@kit.AbilityKit';
import { AudioLabController } from '../main/ets/audio_lab/AudioLabController';
import {
  AudioLabConfig,
  AudioLabControllerDeps,
  AudioLabEvent,
  AudioLabTimerDriver,
  MusicLabLane,
  SfxDuringVoicePolicy,
  SfxLabLane,
  VoiceLabLane,
} from '../main/ets/audio_lab/AudioLabTypes';

class FakeTimerTask {
  id: number = 0;
  ms: number = 0;
  active: boolean = true;
  cb: () => void = (): void => {};
}

class FakeTimer implements AudioLabTimerDriver {
  nextId: number = 1;
  tasks: FakeTimerTask[] = [];

  setTimeout(cb: () => void, ms: number): number {
    const t: FakeTimerTask = new FakeTimerTask();
    t.id = this.nextId;
    this.nextId += 1;
    t.ms = ms;
    t.cb = cb;
    this.tasks.push(t);
    return t.id;
  }

  clearTimeout(id: number): void {
    for (let i = 0; i < this.tasks.length; i++) {
      if (this.tasks[i].id === id) {
        this.tasks[i].active = false;
      }
    }
  }

  activeCount(): number {
    let count: number = 0;
    for (let i = 0; i < this.tasks.length; i++) {
      if (this.tasks[i].active) {
        count += 1;
      }
    }
    return count;
  }

  runLatest(): void {
    for (let i = this.tasks.length - 1; i >= 0; i--) {
      if (this.tasks[i].active) {
        this.tasks[i].active = false;
        this.tasks[i].cb();
        return;
      }
    }
  }
}

class FakeMusicLane implements MusicLabLane {
  preloadedPath: string = '';
  started: number = 0;
  stopped: number = 0;
  disposed: number = 0;
  volumes: number[] = [];
  resumes: number = 0;

  preload(_ctx: common.UIAbilityContext, rawPath: string): Promise<void> {
    this.preloadedPath = rawPath;
    return Promise.resolve();
  }

  startLoop(): void {
    this.started += 1;
  }

  stop(): void {
    this.stopped += 1;
  }

  setVolume(value: number): void {
    this.volumes.push(value);
  }

  resumeOnce(): void {
    this.resumes += 1;
  }

  state(): string {
    return this.started > this.stopped ? 'playing' : 'ready';
  }

  dispose(): Promise<void> {
    this.disposed += 1;
    return Promise.resolve();
  }
}

class FakeSfxLane implements SfxLabLane {

  play(key: string, volume?: number): void {
    this.plays.push(key);
    this.volumes.push(volume !== undefined ? volume : -1);
  }

  dispose(): Promise<void> {
    return Promise.resolve();

}

  describe('AudioLabController', () => {
    it('speakWordLowersMusicThenRestoresOnTimeout', 0, () => {
      const rig: TestRig = new TestRig();

      rig.controller.speakWord('apple');
      expect(rig.voice.spoken[0]).assertEqual('apple');
      expect(rig.music.volumes[0]).assertEqual(0.08);

      expect(rig.music.volumes[1]).assertEqual(0.32);

      expect(rig.sfx.plays.length).assertEqual(1);
      expect(rig.sfx.plays[0]).assertEqual('player_hurt');
      expect(rig.sfx.volumes[0]).assertEqual(0.35);

      expect(rig.timer.activeCount()).assertEqual(0);
      expect(rig.music.disposed).assertEqual(1);
      expect(rig.voice.disposed).assertEqual(1);
    });
  });
}
```

Create `harmonyos/entry/src/main/ets/audio_lab/AudioLabController.ets`:

```ts
import { common } from '@kit.AbilityKit';
import {
  AUDIO_LAB_BGM_RAW_PATH,
  AUDIO_LAB_SFX_KEYS,
  AudioLabConfig,
  AudioLabControllerDeps,
  AudioLabEvent,
  AudioLabObserver,
  AudioLabSnapshot,
  AudioLabTimerDriver,
  MusicLabLane,
  RealAudioLabTimerDriver,
  SfxDuringVoicePolicy,
  SfxLabLane,
  VoiceLabLane,
} from './AudioLabTypes';

const KEY_HIT_NORMAL: string = 'hit_normal';
const KEY_HIT_CRIT: string = 'hit_crit';
const KEY_ANSWER_WRONG: string = 'answer_wrong';
const KEY_PLAYER_HURT: string = 'player_hurt';
const KEY_MONSTER_DEFEAT: string = 'monster_defeat';
const KEY_VICTORY: string = 'victory';
const KEY_DEFEAT: string = 'defeat';

function missingMusicLane(): MusicLabLane {
  throw new Error('AudioLabController missing MusicLabLane dependency');
}

function missingSfxLane(): SfxLabLane {
  throw new Error('AudioLabController missing SfxLabLane dependency');
}

function missingVoiceLane(): VoiceLabLane {
  throw new Error('AudioLabController missing VoiceLabLane dependency');
}

export class AudioLabController {
  private config: AudioLabConfig;
  private observer?: AudioLabObserver;
  private music: MusicLabLane;
  private sfx: SfxLabLane;
  private voice: VoiceLabLane;
  private timer: AudioLabTimerDriver;
  private snap: AudioLabSnapshot = new AudioLabSnapshot();
  private restoreTimer: number = -1;
  private voiceToken: number = 0;

  constructor(config?: AudioLabConfig, observer?: AudioLabObserver, deps?: AudioLabControllerDeps) {
    this.config = config !== undefined ? config : new AudioLabConfig();
    this.observer = observer;
    this.music = deps?.music !== undefined ? deps.music : missingMusicLane();
    this.sfx = deps?.sfx !== undefined ? deps.sfx : missingSfxLane();
    this.voice = deps?.voice !== undefined ? deps.voice : missingVoiceLane();
    this.timer = deps?.timer !== undefined ? deps.timer : new RealAudioLabTimerDriver();
    this.snap.sfxDuringVoicePolicy = this.config.sfxDuringVoicePolicy;
  }

  async enter(ctx: common.UIAbilityContext): Promise<void> {
    if (this.snap.disposed) {
      return;
    }
    await Promise.all([
      this.music.preload(ctx, AUDIO_LAB_BGM_RAW_PATH).catch((err: Error): void => this.recordError('music preload', err)),
      this.sfx.preload(ctx, AUDIO_LAB_SFX_KEYS).catch((err: Error): void => this.recordError('sfx preload', err)),
      this.voice.init().catch((err: Error): void => this.recordError('voice init', err)),
    ]);
    this.emit(AudioLabEvent.Enter);
  }

  async exit(): Promise<void> {
    this.snap.disposed = true;
    this.snap.voiceActive = false;
    this.clearRestoreTimer();
    this.snap.pendingDelayedSfx = '';
    this.music.stop();
    await Promise.all([
      this.music.dispose(),
      this.sfx.dispose(),
    ]);
    this.voice.dispose();
    this.emit(AudioLabEvent.Exit);
  }

  updateConfig(next: AudioLabConfig): void {
    this.config = next;
    this.snap.sfxDuringVoicePolicy = next.sfxDuringVoicePolicy;
    this.emitSnapshot();
  }

  snapshot(): AudioLabSnapshot {
    const out: AudioLabSnapshot = new AudioLabSnapshot();
    out.musicState = this.music.state();
    out.voiceActive = this.snap.voiceActive;
    out.disposed = this.snap.disposed;
    out.musicVolume = this.snap.musicVolume;
    out.lastEvent = this.snap.lastEvent;
    out.lastError = this.snap.lastError;
    out.pendingTimers = this.restoreTimer >= 0 ? 1 : 0;
    out.resumeAttempts = this.snap.resumeAttempts;
    out.pendingDelayedSfx = this.snap.pendingDelayedSfx;
    out.sfxDuringVoicePolicy = this.config.sfxDuringVoicePolicy;
    return out;
  }

  startMusic(): void {
    if (this.snap.disposed || !this.config.musicEnabled) {
      return;
    }
    this.music.setVolume(this.config.musicVolume * this.config.masterVolume);
    this.snap.musicVolume = this.config.musicVolume;
    this.music.startLoop();
    this.emit(AudioLabEvent.StartMusic);
  }

  stopMusic(): void {
    if (this.snap.disposed) {
      return;
    }
    this.music.stop();
    this.emit(AudioLabEvent.StopMusic);
  }

  lowerMusicForVoice(): void {
    this.lowerMusicVolume('voice');
  }

  lowerMusicTemporarily(reason: string, durationMs: number): void {
    if (this.snap.disposed || !this.config.musicEnabled) {
      return;
    }
    this.lowerMusicVolume(reason);
    this.clearRestoreTimer();
    this.restoreTimer = this.timer.setTimeout((): void => {
      this.restoreTimer = -1;
      this.restoreMusicVolume(reason);
    }, durationMs);
  }

  restoreMusicVolume(reason: string): void {
    if (this.snap.disposed || !this.config.musicEnabled) {
      return;
    }
    this.music.setVolume(this.config.musicVolume * this.config.masterVolume);
    this.snap.musicVolume = this.config.musicVolume;
    this.emit(AudioLabEvent.RestoreMusic, reason);
  }

  resumeMusicOnce(reason: string): void {
    if (this.snap.disposed || !this.config.musicEnabled) {
      return;
    }
    this.snap.resumeAttempts += 1;
    this.music.resumeOnce();
    this.emit(AudioLabEvent.ResumeMusicOnce, reason);
  }

  playNormalAttack(): void {
    this.playSfx(KEY_HIT_NORMAL, false, AudioLabEvent.NormalAttack);
  }

  playComboAttack(): void {
    this.lowerMusicTemporarily('combo', this.config.comboLowerDurationMs);
    this.playSfx(KEY_HIT_CRIT, false, AudioLabEvent.ComboAttack);
  }

  playWrongAnswer(): void {
    this.playSfx(KEY_ANSWER_WRONG, false, AudioLabEvent.WrongAnswer);
  }

  playPlayerHurt(): void {
    this.playSfx(KEY_PLAYER_HURT, true, AudioLabEvent.PlayerHurt);
  }

  playMonsterDefeat(): void {
    this.playSfx(KEY_MONSTER_DEFEAT, true, AudioLabEvent.MonsterDefeat);
  }

  playVictory(): void {
    this.playSfx(KEY_VICTORY, true, AudioLabEvent.Victory);
  }

  playDefeat(): void {
    this.playSfx(KEY_DEFEAT, true, AudioLabEvent.Defeat);
  }

  speakWord(word: string): void {
    if (this.snap.disposed || !this.config.voiceEnabled) {
      return;
    }
    const trimmed: string = word.trim();
    if (trimmed.length === 0 || !this.voice.isAvailable()) {
      return;
    }
    this.voiceToken += 1;
    const token: number = this.voiceToken;
    this.snap.voiceActive = true;
    this.lowerMusicForVoice();
    this.voice.speak(trimmed);
    this.emit(AudioLabEvent.SpeakWord, trimmed);
    this.clearRestoreTimer();
    this.restoreTimer = this.timer.setTimeout((): void => this.finishVoice(token), this.config.voiceLowerDurationMs);
  }

  demoSpeakOverMusic(word: string): void {
    this.startMusic();
    this.speakWord(word);
  }

  demoComboOverMusic(): void {
    this.startMusic();
    this.playComboAttack();
  }

  demoWrongAnswerSequence(): void {
    this.playWrongAnswer();
    this.playPlayerHurt();
  }

  demoVictorySequence(): void {
    this.stopMusic();
    this.playVictory();
  }

  demoSfxDuringVoice(word: string): void {
    this.startMusic();
    this.speakWord(word);
    this.playNormalAttack();
    this.playComboAttack();
    this.playPlayerHurt();
  }

  private finishVoice(token: number): void {
    if (this.snap.disposed || token !== this.voiceToken) {
      return;
    }
    this.restoreTimer = -1;
    this.snap.voiceActive = false;
    this.restoreMusicVolume('voice-timeout');
    if (this.config.sfxDuringVoicePolicy === SfxDuringVoicePolicy.DelayUntilVoiceEnds
      && this.snap.pendingDelayedSfx.length > 0) {
      const delayed: string = this.snap.pendingDelayedSfx;
      this.snap.pendingDelayedSfx = '';
      this.sfx.play(delayed, this.config.sfxVolume * this.config.masterVolume);
      this.emit(AudioLabEvent.SfxDelayedPlayed, delayed);
    }
    if (this.config.resumeMusicAfterVoice) {
      this.resumeMusicOnce('voice-timeout');
    }
  }

  private playSfx(key: string, critical: boolean, event: AudioLabEvent): void {
    if (this.snap.disposed || !this.config.sfxEnabled) {
      return;
    }
    if (!this.snap.voiceActive) {
      this.sfx.play(key, this.config.sfxVolume * this.config.masterVolume);
      this.emit(event, key);
      return;
    }
    if (this.config.sfxDuringVoicePolicy === SfxDuringVoicePolicy.PlayFull) {
      this.sfx.play(key, this.config.sfxVolume * this.config.masterVolume);
      this.emit(event, key);
      return;
    }
    if (this.config.sfxDuringVoicePolicy === SfxDuringVoicePolicy.LowerVolume) {
      this.sfx.play(key, this.config.sfxDuringVoiceVolume * this.config.masterVolume);
      this.emit(AudioLabEvent.SfxLoweredForVoice, key);
      return;
    }
    if (this.config.sfxDuringVoicePolicy === SfxDuringVoicePolicy.SuppressNonCritical) {
      if (critical) {
        this.sfx.play(key, this.config.sfxDuringVoiceVolume * this.config.masterVolume);
        this.emit(AudioLabEvent.SfxLoweredForVoice, key);
      } else {
        this.emit(AudioLabEvent.SfxSuppressedForVoice, key);
      }
      return;
    }
    if (critical) {
      this.sfx.play(key, this.config.sfxDuringVoiceVolume * this.config.masterVolume);
      this.emit(AudioLabEvent.SfxLoweredForVoice, key);
    } else {
      this.snap.pendingDelayedSfx = key;
      this.emit(AudioLabEvent.SfxDelayedForVoice, key);
    }
  }

  private lowerMusicVolume(reason: string): void {
    if (this.snap.disposed || !this.config.musicEnabled) {
      return;
    }
    this.music.setVolume(this.config.musicLoweredVolume * this.config.masterVolume);
    this.snap.musicVolume = this.config.musicLoweredVolume;
    this.emit(AudioLabEvent.LowerMusic, reason);
  }

  private clearRestoreTimer(): void {
    if (this.restoreTimer >= 0) {
      this.timer.clearTimeout(this.restoreTimer);
      this.restoreTimer = -1;
    }
  }

  private recordError(context: string, err: Error): void {
    this.snap.lastError = `${context}: ${err.message}`;
    this.emit(AudioLabEvent.Error, this.snap.lastError);
  }

  private emit(event: AudioLabEvent, detail?: string): void {
    this.snap.lastEvent = event;
    if (this.observer !== undefined) {
      try {
        this.observer.onEvent(event, detail);
        this.observer.onSnapshot(this.snapshot());
      } catch (err) {
        console.error(`AudioLabController observer failed: ${JSON.stringify(err as Error)}`);
      }
    }
  }

  private emitSnapshot(): void {
    if (this.observer !== undefined) {
      try {
        this.observer.onSnapshot(this.snapshot());
      } catch (err) {
        console.error(`AudioLabController observer snapshot failed: ${JSON.stringify(err as Error)}`);
      }
    }
  }
}
```

```ts
export class RealMusicLabLane implements MusicLabLane {
  private player?: media.AVPlayer;
  private ready: boolean = false;
  private muted: boolean = false;
  private currentState: string = 'idle';
  private currentVolume: number = 0.32;

  async preload(ctx: common.UIAbilityContext, rawPath: string): Promise<void> {
    if (this.player !== undefined) {
      return;
    }
    try {
      const raw = await ctx.resourceManager.getRawFd(rawPath);
      const player: media.AVPlayer = await media.createAVPlayer();
      this.player = player;
      player.on('error', (err: BusinessError): void => {
        this.currentState = 'error';
        this.muted = true;
        console.error(`RealMusicLabLane error: ${JSON.stringify(err)}`);
      });

        this.currentState = state;
        if (state === 'initialized') {
          player.prepare().catch((err: BusinessError): void => {
            this.muted = true;
            console.error(`RealMusicLabLane prepare failed: ${JSON.stringify(err)}`);
          });

          this.ready = true;
          player.setVolume(this.currentVolume);
        } else if (state === 'completed') {
          player.seek(0);
          player.play().catch((err: BusinessError): void => {
            console.error(`RealMusicLabLane loop replay failed: ${JSON.stringify(err)}`);
          });
        }
      });
      player.fdSrc = {
        fd: raw.fd,
        offset: raw.offset,
        length: raw.length,
      };
    } catch (err) {
      this.currentState = 'error';
      this.muted = true;
      console.error(`RealMusicLabLane preload failed: ${JSON.stringify(err as BusinessError)}`);
    }
  }

  startLoop(): void {
    if (this.muted || !this.ready || this.player === undefined) {
      return;
    }
    this.player.play().catch((err: BusinessError): void => {
      console.error(`RealMusicLabLane play failed: ${JSON.stringify(err)}`);
    });
  }

  stop(): void {
    if (this.player === undefined) {
      return;
    }
    this.player.pause().catch((err: BusinessError): void => {
      console.error(`RealMusicLabLane pause failed: ${JSON.stringify(err)}`);
    });
  }

  setVolume(value: number): void {
    this.currentVolume = Math.max(0, Math.min(1, value));
    if (this.player !== undefined) {
      this.player.setVolume(this.currentVolume);
    }
  }

  resumeOnce(): void {
    this.startLoop();
  }

  state(): string {
    return this.currentState;
  }

  async dispose(): Promise<void> {

    this.player = undefined;
    this.ready = false;
    this.muted = true;

        console.error(`RealMusicLabLane release failed: ${JSON.stringify(err)}`);
      });
    }
  }
```

```ts
import media from '@ohos.multimedia.media';
import { BusinessError } from '@ohos.base';
import { common } from '@kit.AbilityKit';
import { SfxLabLane } from '../AudioLabTypes';

const SFX_RAW_PATHS: Record<string, string> = {
  'hit_normal': 'sound/hit_normal.ogg',
  'hit_crit': 'sound/hit_crit.ogg',
  'answer_wrong': 'sound/answer_wrong.ogg',
  'player_hurt': 'sound/player_hurt.ogg',
  'monster_defeat': 'sound/monster_defeat.ogg',
  'victory': 'sound/victory.ogg',
  'defeat': 'sound/defeat.ogg',
};

class LabSfxSlot {
  player?: media.AVPlayer;
  ready: boolean = false;
  muted: boolean = false;
}

export class RealSfxLabLane implements SfxLabLane {
  private slots: Map<string, LabSfxSlot> = new Map<string, LabSfxSlot>();
  private disposed: boolean = false;

  async preload(ctx: common.UIAbilityContext, keys: string[]): Promise<void> {
    const tasks: Promise<void>[] = [];
    for (let i = 0; i < keys.length; i++) {
      const key: string = keys[i];
      if (this.slots.has(key)) {
        continue;
      }
      const slot: LabSfxSlot = new LabSfxSlot();
      this.slots.set(key, slot);
      tasks.push(this.loadOne(ctx, key, slot));
    }
    await Promise.all(tasks);
  }

  private async loadOne(ctx: common.UIAbilityContext, key: string, slot: LabSfxSlot): Promise<void> {
    const rawPath: string | undefined = SFX_RAW_PATHS[key];
    if (rawPath === undefined) {
      slot.muted = true;
      return;
    }
    try {
      const raw = await ctx.resourceManager.getRawFd(rawPath);
      const player: media.AVPlayer = await media.createAVPlayer();
      slot.player = player;
      player.on('error', (err: BusinessError): void => {
        slot.muted = true;
        console.error(`RealSfxLabLane[${key}] error: ${JSON.stringify(err)}`);
      });
      player.on('stateChange', (state: string): void => {
        if (state === 'initialized') {
          player.prepare().catch((err: BusinessError): void => {
            slot.muted = true;
            console.error(`RealSfxLabLane[${key}] prepare failed: ${JSON.stringify(err)}`);
          });
        } else if (state === 'prepared') {
          slot.ready = true;
        }
      });
      player.fdSrc = {
        fd: raw.fd,
        offset: raw.offset,
        length: raw.length,
      };
    } catch (err) {
      slot.muted = true;
      console.error(`RealSfxLabLane[${key}] preload failed: ${JSON.stringify(err as BusinessError)}`);
    }
  }

  play(key: string, volume?: number): void {
    if (this.disposed) {
      return;
    }
    const slot: LabSfxSlot | undefined = this.slots.get(key);
    if (slot === undefined || slot.muted || !slot.ready || slot.player === undefined) {
      return;
    }
    const effective: number = volume !== undefined ? Math.max(0, Math.min(1, volume)) : 1;
    slot.player.setVolume(effective);
    slot.player.seek(0);
    slot.player.play().catch((err: BusinessError): void => {
      console.error(`RealSfxLabLane[${key}] play failed: ${JSON.stringify(err)}`);
    });
  }

  async dispose(): Promise<void> {
    this.disposed = true;
    const releases: Promise<void>[] = [];
    this.slots.forEach((slot: LabSfxSlot): void => {
      const player: media.AVPlayer | undefined = slot.player;
      slot.player = undefined;
      slot.ready = false;
      slot.muted = true;
      if (player !== undefined) {
        releases.push(player.release().catch((err: BusinessError): void => {
          console.error(`RealSfxLabLane release failed: ${JSON.stringify(err)}`);
        }));
      }
    });
    this.slots.clear();
    await Promise.all(releases);
  }
}
```
