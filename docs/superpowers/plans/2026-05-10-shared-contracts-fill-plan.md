```python
from __future__ import annotations

import hashlib
import json
from pathlib import Path

from app.main import app


REPO_ROOT = Path(__file__).resolve().parents[2]
OUT_DIR = REPO_ROOT / "shared" / "contracts" / "openapi"
OPENAPI_PATH = OUT_DIR / "happyword-api.openapi.json"
PATHS_PATH = OUT_DIR / "happyword-api.paths.txt"
SHA_PATH = OUT_DIR / "happyword-api.sha256"


def normalized_openapi() -> dict[str, object]:
    data = app.openapi()
    data["info"]["x-contract-source"] = "server/app/main.py"
    data["info"]["x-contract-note"] = (
        "Generated by tools/contracts/export_openapi.py. Do not edit by hand."
    )
    return data


def write_contracts() -> None:
    OUT_DIR.mkdir(parents=True, exist_ok=True)
    data = normalized_openapi()
    payload = json.dumps(data, ensure_ascii=False, indent=2, sort_keys=True) + "\n"
    OPENAPI_PATH.write_text(payload, encoding="utf-8")
    PATHS_PATH.write_text(
        "\n".join(sorted(data["paths"].keys())) + "\n",
        encoding="utf-8",
    )
    SHA_PATH.write_text(
        hashlib.sha256(payload.encode("utf-8")).hexdigest() + "\n",
        encoding="utf-8",
    )


if __name__ == "__main__":
    write_contracts()
```

from __future__ import annotations

import hashlib
import json
from pathlib import Path

from app.main import app


REPO_ROOT = Path(__file__).resolve().parents[2]
OPENAPI_PATH = REPO_ROOT / "shared" / "contracts" / "openapi" / "happyword-api.openapi.json"
PATHS_PATH = REPO_ROOT / "shared" / "contracts" / "openapi" / "happyword-api.paths.txt"
SHA_PATH = REPO_ROOT / "shared" / "contracts" / "openapi" / "happyword-api.sha256"


def main() -> None:
    generated = app.openapi()
    generated["info"]["x-contract-source"] = "server/app/main.py"
    generated["info"]["x-contract-note"] = (
        "Generated by tools/contracts/export_openapi.py. Do not edit by hand."
    )
    payload = json.dumps(generated, ensure_ascii=False, indent=2, sort_keys=True) + "\n"
    recorded = OPENAPI_PATH.read_text(encoding="utf-8")
    if payload != recorded:
        raise SystemExit(
            "OpenAPI contract is stale. Run: cd server && uv run python ../tools/contracts/export_openapi.py"
        )
    expected_paths = "\n".join(sorted(generated["paths"].keys())) + "\n"
    if PATHS_PATH.read_text(encoding="utf-8") != expected_paths:
        raise SystemExit("OpenAPI paths index is stale.")
    expected_sha = hashlib.sha256(payload.encode("utf-8")).hexdigest() + "\n"
    if SHA_PATH.read_text(encoding="utf-8") != expected_sha:
        raise SystemExit("OpenAPI SHA is stale.")


if __name__ == "__main__":
    main()
```

# WordMagicGame Shared Contracts

`shared/contracts/` is the cross-platform contract home for HarmonyOS, iOS, Android, and the FastAPI server.

## What Belongs Here

- Generated OpenAPI snapshots for JSON APIs.
- Human-readable domain indexes for app-facing and admin-facing API groups.
- Protocol notes for sync flows that need more detail than an endpoint list.
- Error envelope and error-code documentation.
- Links to golden fixtures under `shared/fixtures/`.

## What Does Not Belong Here

- Shared client runtime code.
- Generated Swift, Kotlin, ArkTS, or Python clients.
- UI components or business logic imported by native clients.
- Server implementation code.

The server source of truth is `server/app/main.py` plus the included routers and Pydantic schemas.

Generated files:

- `openapi/happyword-api.openapi.json`

# Error Envelope Contract

FastAPI validation errors still use FastAPI's default `detail` shape unless a router explicitly raises a structured detail.

Project-specific errors should use:

```json
{
  "detail": {
    "error": {
      "code": "WORD_NOT_FOUND",
      "message": "Word not found"
    }
  }
}
```

Native clients should parse in this order:

1. `detail.error.code` and `detail.error.message`
2. string `detail`
3. FastAPI validation list `detail[]`
4. fallback to HTTP status text

No client should branch only on localized `message` text.
```

| global | `GET /api/v1/public/global-packs/latest.json` | anonymous | ETag + fetchedAt |
| family | `GET /api/v1/family/{family_id}/family-packs/latest.json` | device token | ETag + fetchedAt |

## Status Handling

| HTTP status | Meaning | Client action |
| --- | --- | --- |
| 200 | New payload available | Replace cached layer and record `ETag`. |
| 204 | No packs published for this layer | Clear cached layer for the requesting context. |
| 304 | Cache still current | Keep cached layer. |
| 401 / 403 | Device auth invalid | Keep local playable fallback; surface binding problem in parent/account UI. |
| 410 | Binding or token gone | Clear cloud credentials after user confirmation or server-directed unbind flow. |
| 5xx / network | Temporary failure | Keep cached layer and built-in fallback. |

## Merge Semantics

Client effective library is `family > global > built-in` by `pack_id`. `shared/` must not provide a runtime merge implementation; native clients implement the merge and verify against fixtures.
```
