# V0.8.1 Parent Vocabulary Management Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the parent-side vocabulary management loop: create family packs, batch edit/import draft words, publish/rollback, and let bound child devices fetch a tenant-scoped global + family merged pack.

**Architecture:** Reuse the existing family-pack model and API as the source of truth. Add focused service helpers for draft validation, batch upsert, import-to-draft, and global + family merging. Add parent HTML routes/templates on top of the existing `/parent/` shell, keeping JSON APIs testable and UI pages thin.

**Tech Stack:** FastAPI, Beanie/MongoDB, Pydantic v2, Jinja2 templates, htmx/Tailwind parent web shell, pytest + httpx ASGI tests, `uv`.

---

```python
async def batch_upsert_draft_words(
    *,
    definition: FamilyPackDefinition,
    rows: list[dict[str, Any]],
    parent_user_id: str,
) -> tuple[FamilyPackDraft, list[dict[str, Any]]]:
    errors: list[dict[str, Any]] = []
    draft = await get_or_create_draft(
        definition=definition, parent_user_id=parent_user_id
    )

    for idx, row in enumerate(rows):
        word_id = str(row.get("word_id", ""))
        payload = {k: v for k, v in row.items() if k != "word_id"}
        try:
            draft = await upsert_draft_word(
                definition=definition,
                word_id=word_id,
                payload=payload,
                parent_user_id=parent_user_id,
            )
        except FamilyPackError as exc:
            errors.append(
                {
                    "row_index": idx,
                    "word_id": word_id,
                    "code": exc.code,
                    "message": str(exc),
                }
            )
    return draft, errors
```

```python
from __future__ import annotations

import re
from typing import Any

from app.models.family_pack_definition import FamilyPackDefinition
from app.services import family_pack_service
from app.services.lesson_service import extract_lesson_payload, upload_lesson_image


def _slug_word(word: str) -> str:
    cleaned = re.sub(r"[^a-z0-9]+", "-", word.strip().lower()).strip("-")
    return cleaned or "word"


async def upload_family_pack_image(payload: bytes, mime: str) -> str:
    return await upload_lesson_image(payload, mime)


async def extract_family_pack_image(payload: bytes, mime: str) -> tuple[str, dict[str, Any]]:
    return await extract_lesson_payload(payload, mime)


def extracted_words_to_rows(*, family_id: str, extracted: dict[str, Any]) -> list[dict[str, Any]]:
    category_obj = extracted.get("category")
    fallback_category = "custom"
    if isinstance(category_obj, dict) and isinstance(category_obj.get("id"), str):
        fallback_category = category_obj["id"]
    words_raw = extracted.get("words", [])
    if not isinstance(words_raw, list):
        return []

    prefix = family_pack_service.CustomIdContract(family_id=family_id).prefix
    rows: list[dict[str, Any]] = []
    for item in words_raw:
        if not isinstance(item, dict):
            continue
        word = item.get("word")
        meaning = item.get("meaningZh") or item.get("meaning_zh")
        if not isinstance(word, str) or not isinstance(meaning, str):
            continue
        category = item.get("category")
        difficulty = item.get("difficulty", 1)
        rows.append(
            {
                "word_id": f"{prefix}{_slug_word(word)}",
                "source": "custom",
                "word": word.strip(),
                "meaning_zh": meaning.strip(),
                "category": category if isinstance(category, str) else fallback_category,
                "difficulty": difficulty if isinstance(difficulty, int) else 1,
            }
        )
    return rows


async def import_image_to_draft(
    *,
    definition: FamilyPackDefinition,
    payload: bytes,
    mime: str,
    parent_user_id: str,
) -> tuple[str, str, int, object, list[dict[str, Any]]]:
    source_image_url = await upload_family_pack_image(payload, mime)
    model_name, extracted = await extract_family_pack_image(payload, mime)
    rows = extracted_words_to_rows(family_id=definition.family_id, extracted=extracted)
    draft, errors = await family_pack_service.batch_upsert_draft_words(
        definition=definition,
        rows=rows,
        parent_user_id=parent_user_id,
    )
    return source_image_url, model_name, len(rows) - len(errors), draft, errors
```

```python
@router.post(
    "/{pack_id}/import-image",
    response_model=FamilyPackImportImageOut,
    status_code=status.HTTP_201_CREATED,
)
async def import_image_to_pack(
    pack_id: str,
    image: UploadFile = File(...),
    user: User = Depends(current_parent_user),
) -> FamilyPackImportImageOut:
    mime = (image.content_type or "").lower()
    if mime not in _ACCEPTED_IMAGE_MIME:
        raise HTTPException(
            status_code=status.HTTP_415_UNSUPPORTED_MEDIA_TYPE,
            detail={"error": {"code": "UNSUPPORTED_MEDIA_TYPE", "message": f"Unsupported image type {mime!r}"}},
        )
    payload = await image.read()
    if not payload:
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail={"error": {"code": "EMPTY_BODY", "message": "Uploaded image is empty"}},
        )
    if len(payload) > _MAX_IMAGE_BYTES:
        raise HTTPException(
            status_code=status.HTTP_413_REQUEST_ENTITY_TOO_LARGE,
            detail={"error": {"code": "IMAGE_TOO_LARGE", "message": f"Image is {len(payload)} bytes"}},
        )

    definition = await _load_definition_or_404(pack_id, user.family_id or "")
    try:
        source_image_url, model_name, imported_count, draft, errors = (
            await family_pack_import_service.import_image_to_draft(
                definition=definition,
                payload=payload,
                mime=mime,
                parent_user_id=user.username,
            )
        )
    except LlmConfigError as exc:
        raise HTTPException(
            status_code=status.HTTP_503_SERVICE_UNAVAILABLE,
            detail={"error": {"code": "LLM_NOT_CONFIGURED", "message": str(exc)}},
        ) from exc
    except LlmCallError as exc:
        raise HTTPException(
            status_code=status.HTTP_502_BAD_GATEWAY,
            detail={"error": {"code": "LLM_CALL_FAILED", "message": str(exc)}},
        ) from exc

    return FamilyPackImportImageOut(
        pack_id=pack_id,
        source_image_url=source_image_url,
        imported_count=imported_count,
        model=model_name,
        draft=_serialize_draft(draft),
        errors=[FamilyPackDraftWordBatchError(**e) for e in errors],
    )
```

```python
@dataclass(frozen=True)
class ChildMergedVocabulary:
    schema_version: int
    family_id: str
    global_version: int
    family_versions: dict[str, int]
    words: list[dict[str, Any]]
    slices: list[MergedSlice]
    etag: str


def _merge_words(global_words: list[dict[str, Any]], family_slices: list[MergedSlice]) -> list[dict[str, Any]]:
    merged: dict[str, dict[str, Any]] = {}
    hidden: set[str] = set()
    for word in global_words:
        word_id = word.get("id")
        if isinstance(word_id, str):
            merged[word_id] = dict(word)
    for family_slice in family_slices:
        for word in family_slice.words:
            word_id = word.get("id")
            if not isinstance(word_id, str):
                continue
            if word.get("hidden") is True:
                hidden.add(word_id)
                merged.pop(word_id, None)
                continue
            if word_id in hidden:
                hidden.remove(word_id)
            merged[word_id] = dict(word)
    return [merged[k] for k in sorted(merged)]


def _child_etag(*, global_version: int, family_pairs: list[tuple[str, int]]) -> str:
    family_part = "|".join(f"{pid}:{ver}" for pid, ver in sorted(family_pairs))
    digest = hashlib.sha256(f"global:{global_version}|family:{family_part}".encode("utf-8")).hexdigest()
    return f'"{digest}"'


async def collect_child_vocabulary(*, family_id: str) -> ChildMergedVocabulary:
    global_version, global_payload = await pack_service.get_current_pack_payload()
    family_slices, _ = await collect_merged(family_id=family_id)
    family_versions = {s.pack_id: s.version for s in family_slices}
    global_words = list(global_payload.get("words", []))
    words = _merge_words(global_words=global_words, family_slices=family_slices)
    family_pairs = [(s.pack_id, s.version) for s in family_slices]
    schema_version = max(
        [int(global_payload.get("schema_version", GLOBAL_PACK_SCHEMA_VERSION))]
        + [s.schema_version for s in family_slices]
    )
    return ChildMergedVocabulary(
        schema_version=schema_version,
        family_id=family_id,
        global_version=global_version,
        family_versions=family_versions,
        words=words,
        slices=family_slices,
        etag=_child_etag(global_version=global_version, family_pairs=family_pairs),
    )
```

```python
    x_family_id: str | None = Header(default=None, alias="X-Family-Id"),
```

```python
        request,

        {"user": user, "error": ""},
    )

    name: str = Form(...),
    description: str = Form(""),
    user: User = Depends(current_parent_user),
) -> RedirectResponse:
    definition = await svc.create_definition(
        family_id=user.family_id or "",
        name=name,
        description=description or None,
        parent_user_id=user.username,
    )
    return RedirectResponse(url=f"/parent/packs/{definition.pack_id}", status_code=303)

async def detail_page(
    request: Request,

        request,
        "parent/packs/detail.html",
        {
            "user": user,
            "definition": _serialize_definition(definition),
            "pointer": pointer,
            "current_pack": pack,
            "draft": _serialize_draft(draft),
```

```python
@router.get("/{pack_id}/import", response_class=HTMLResponse, response_model=None)
async def import_page(
    request: Request,
    pack_id: str,
    user: User = Depends(current_parent_user),
) -> HTMLResponse | RedirectResponse:
    definition = await _load_definition_or_redirect(pack_id, user)
    if definition is None:

    return templates.TemplateResponse(
        request,

        {"user": user, "definition": _serialize_definition(definition)},
    )


@router.post("/{pack_id}/publish", response_model=None)
async def publish_page(
    pack_id: str,
    notes: str = Form(""),
    user: User = Depends(current_parent_user),
) -> RedirectResponse:
    definition = await svc.get_definition_for_family(
        pack_id=pack_id,
        family_id=user.family_id or "",
    )
    await svc.publish(definition=definition, parent_user_id=user.username, notes=notes or None)
    return RedirectResponse(url=f"/parent/packs/{pack_id}", status_code=303)


@router.get("/{pack_id}/versions", response_class=HTMLResponse, response_model=None)
async def versions_page(
    request: Request,

    return templates.TemplateResponse(
```
