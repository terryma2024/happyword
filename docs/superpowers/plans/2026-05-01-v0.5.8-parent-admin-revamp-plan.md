# V0.5.8 Parent Admin Console + Photo Lesson Import — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the JWT-gated AdminConsole with a parent-PIN-only ParentAdmin page that locks portrait, hosts a photo→LLM→review→approve flow for adding textbook words, and keeps a separate one-tap pack publish.

**Architecture:** Server drops `Depends(current_admin_user)` from 8 admin routers (open admin, V0.6 reintroduces auth). Client renames `AdminConsole*` → `ParentAdmin*` end-to-end (page, route, ids, class). Photo upload via `picker.PhotoViewPicker` + `cameraPicker`, multipart POST to existing `/api/v1/admin/lessons/import`, server returns a `LessonImportDraft`, client pushes a new `LessonDraftReviewPage` for parent edit/approve. Per-window portrait orientation lock kept inside the admin flow only.

**Tech Stack:** HarmonyOS NEXT / ArkTS / ArkUI; `@kit.NetworkKit` for HTTP, `@kit.CoreFileKit` `picker.PhotoViewPicker` for gallery, `@kit.CameraKit` `cameraPicker` for camera, `@kit.ArkUI` `window.setPreferredOrientation` for orientation; FastAPI + Beanie + OpenAI Vision (`gpt-4o`) on the server side; Hypium for tests.

**Spec:** [`docs/superpowers/specs/2026-05-01-v0.5.8-parent-admin-revamp-design.md`](../specs/2026-05-01-v0.5.8-parent-admin-revamp-design.md). All design decisions live there; this plan executes them.

**Branch:** `feat/v0.5.8-parent-admin-revamp` (already checked out).

```typescript
import { window } from '@kit.ArkUI';
import { common } from '@kit.AbilityKit';

/**
 * V0.5.8: per-window portrait orientation lock for the parent admin
 * flow. Battle and home stay landscape (the app default), but the
 * admin pages prefer portrait because they are vertical lists of
 * forms / words.
 *
 * Lifecycle policy (see spec §6.2):
 *   - ParentAdminPage:        aboutToAppear + onPageShow → lockPortrait
 *                             aboutToDisappear           → restoreAutoLandscape
 *   - LessonDraftReviewPage:  aboutToAppear + onPageShow → lockPortrait
 *                             (no restore — outer page handles it)
 *
 * Helpers swallow adapter errors so a buggy emulator or runtime
 * denial still leaves the page usable (just in landscape).
 */

export type OrientationName = 'PORTRAIT' | 'AUTO_ROTATION_LANDSCAPE';

export interface IOrientationAdapter {
  setPreferredOrientation(name: OrientationName): Promise<void>;
}

export class RealOrientationAdapter implements IOrientationAdapter {
  private ctx: common.UIAbilityContext;

  constructor(ctx: common.UIAbilityContext) {
    this.ctx = ctx;
  }

  async setPreferredOrientation(name: OrientationName): Promise<void> {
    const win: window.Window = await window.getLastWindow(this.ctx);
    const target: window.Orientation = name === 'PORTRAIT'
      ? window.Orientation.PORTRAIT
      : window.Orientation.AUTO_ROTATION_LANDSCAPE;
    await win.setPreferredOrientation(target);
  }
}

export async function lockPortrait(adapter: IOrientationAdapter): Promise<void> {
  try {
    await adapter.setPreferredOrientation('PORTRAIT');
  } catch (_e) {
    // Swallow — see spec §6.3 fallback.
  }
}

export async function restoreAutoLandscape(adapter: IOrientationAdapter): Promise<void> {
  try {
    await adapter.setPreferredOrientation('AUTO_ROTATION_LANDSCAPE');
  } catch (_e) {
    // Swallow.
  }
}
```

```typescript
import { buffer } from '@kit.ArkTS';

/**
 * V0.5.8: hand-rolled multipart/form-data builder for the single-file
 * lesson-import upload. Hand-rolled because @kit.NetworkKit's http
 * request takes either a string body (for text) or an ArrayBuffer
 * (for binary), and FastAPI's UploadFile expects an RFC 7578
 * multipart envelope. We assemble the envelope as one ArrayBuffer
 * so the http layer can hand it straight to the wire.
 *
 * Field name is fixed to "image" — that's what the server's
 * @router.post("/lessons/import", image: UploadFile = File(...))
 * expects.
 */

export class MultipartResult {
  body: ArrayBuffer = new ArrayBuffer(0);
  contentType: string = '';
}

const FIELD_NAME: string = 'image';

export function buildSingleImageMultipart(
  fileBytes: ArrayBuffer,
  mime: string,
  filename: string,
  boundary: string,
): MultipartResult {
  const header: string =
    `--${boundary}\r\n` +
    `Content-Disposition: form-data; name="${FIELD_NAME}"; filename="${escapeFilename(filename)}"\r\n` +
    `Content-Type: ${mime}\r\n\r\n`;
  const trailer: string = `\r\n--${boundary}--\r\n`;

  const headerBytes: Uint8Array = new Uint8Array(buffer.from(header, 'utf-8').buffer);
  const trailerBytes: Uint8Array = new Uint8Array(buffer.from(trailer, 'utf-8').buffer);
  const fileView: Uint8Array = new Uint8Array(fileBytes);

  const total: number = headerBytes.length + fileView.length + trailerBytes.length;
  const out: Uint8Array = new Uint8Array(total);
  out.set(headerBytes, 0);
  out.set(fileView, headerBytes.length);
  out.set(trailerBytes, headerBytes.length + fileView.length);

  const result: MultipartResult = new MultipartResult();
  result.body = out.buffer;
  result.contentType = `multipart/form-data; boundary=${boundary}`;
  return result;
}

/** Escape backslashes and quotes so the filename can sit inside a
 *  quoted form-data Content-Disposition value. */
function escapeFilename(name: string): string {
  let s: string = '';
  for (let i: number = 0; i < name.length; i++) {
    const c: string = name[i];
    if (c === '"' || c === '\\') {
      s += '\\';
    }
    s += c;
  }
  return s;
}
```

```typescript
import { picker } from '@kit.CoreFileKit';
import { cameraPicker as camPicker } from '@kit.CameraKit';
import { fileIo as fs } from '@kit.CoreFileKit';
import { common } from '@kit.AbilityKit';
import { PickedImage } from './ParentApiClient';

/**
 * V0.5.8: photo source for the parent admin lesson-import flow.
 * Two seams: IPhotoPickerAdapter (selects a uri via system UI) and
 * IPhotoFileReader (reads bytes off the uri). Tests stub both.
 *
 * MIME is sniffed from the uri extension because the system pickers
 * don't return a separate MIME field. Server validates again.
 */

export interface PickedFileRef {
  uri: string;
}

export interface IPhotoPickerAdapter {
  selectGallery(): Promise<PickedFileRef | null>;
  selectCamera(): Promise<PickedFileRef | null>;
}

export interface IPhotoFileReader {
  readAsArrayBuffer(uri: string): Promise<ArrayBuffer>;
  sizeOf(uri: string): Promise<number>;
}

export class RealPhotoPickerAdapter implements IPhotoPickerAdapter {
  private ctx: common.UIAbilityContext;

  constructor(ctx: common.UIAbilityContext) {
    this.ctx = ctx;
  }

  async selectGallery(): Promise<PickedFileRef | null> {
    const opts: picker.PhotoSelectOptions = new picker.PhotoSelectOptions();
    opts.MIMEType = picker.PhotoViewMIMETypes.IMAGE_TYPE;
    opts.maxSelectNumber = 1;
    const photoPicker: picker.PhotoViewPicker = new picker.PhotoViewPicker();
    const res: picker.PhotoSelectResult = await photoPicker.select(opts);
    if (!res.photoUris || res.photoUris.length === 0) {
      return null;
    }
    return { uri: res.photoUris[0] };
  }

  async selectCamera(): Promise<PickedFileRef | null> {
    const opts: camPicker.PickerProfile = {
      cameraPosition: camPicker.CameraPosition.CAMERA_POSITION_BACK,
    };
    const res: camPicker.PickerResult = await camPicker.pick(
      this.ctx,
      [camPicker.PickerMediaType.PHOTO],
      opts,
    );
    if (!res || res.resultCode !== 0 || !res.resultUri) {
      return null;
    }
    return { uri: res.resultUri };
  }
}

export class RealPhotoFileReader implements IPhotoFileReader {
  async readAsArrayBuffer(uri: string): Promise<ArrayBuffer> {
    let f: fs.File | null = null;
    try {
      f = await fs.open(uri, fs.OpenMode.READ_ONLY);
      const stat: fs.Stat = await fs.stat(f.fd);
      const buf: ArrayBuffer = new ArrayBuffer(stat.size);
      await fs.read(f.fd, buf);
      return buf;
    } finally {
      if (f !== null) {
        try { await fs.close(f.fd); } catch (_e) {}
      }
    }
  }

  async sizeOf(uri: string): Promise<number> {
    let f: fs.File | null = null;
    try {
      f = await fs.open(uri, fs.OpenMode.READ_ONLY);
      const stat: fs.Stat = await fs.stat(f.fd);
      return stat.size;
    } finally {
      if (f !== null) {
        try { await fs.close(f.fd); } catch (_e) {}
      }
    }
  }
}

export class LessonImagePicker {
  private picker: IPhotoPickerAdapter;
  private reader: IPhotoFileReader;

  constructor(p: IPhotoPickerAdapter, r: IPhotoFileReader) {
    this.picker = p;
    this.reader = r;
  }

  async pickFromGallery(): Promise<PickedImage | null> {
    const ref: PickedFileRef | null = await this.picker.selectGallery();
    return this.materialize(ref);
  }

  async pickFromCamera(): Promise<PickedImage | null> {
    const ref: PickedFileRef | null = await this.picker.selectCamera();
    return this.materialize(ref);
  }

  private async materialize(ref: PickedFileRef | null): Promise<PickedImage | null> {
    if (ref === null) {
      return null;
    }
    const filename: string = filenameOf(ref.uri);
    const mime: string = sniffMime(filename);
    if (mime === '') {
      return null;
    }
    const bytes: ArrayBuffer = await this.reader.readAsArrayBuffer(ref.uri);
    const size: number = bytes.byteLength;

    const out: PickedImage = new PickedImage();
    out.bytes = bytes;
    out.mime = mime;
    out.filename = filename;
    out.sizeBytes = size;
    return out;
  }
}

function filenameOf(uri: string): string {
  const i: number = uri.lastIndexOf('/');
  return (i >= 0) ? uri.substring(i + 1) : uri;
}

function sniffMime(filename: string): string {
  const lower: string = filename.toLowerCase();
  if (lower.endsWith('.jpg') || lower.endsWith('.jpeg')) return 'image/jpeg';
  if (lower.endsWith('.png')) return 'image/png';
  if (lower.endsWith('.webp')) return 'image/webp';
  return '';
}
```

```typescript
private async onApprove(): Promise<void> {
  if (this.draft === null) return;
  const kept: LessonExtractedWord[] = [];
  for (let i: number = 0; i < this.edits.length; i++) {
    if (this.keepFlags[i]) kept.push(this.edits[i]);
  }
  if (kept.length === 0) {
    this.error = '请至少保留一个单词';
    return;
  }
  this.busy = true;
  this.error = '';
  try {
    const edited: LessonExtractedDto = new LessonExtractedDto();
    edited.lesson_title = this.draft.extracted.lesson_title;
    edited.category_id = this.draft.extracted.category_id;
    edited.category_label_zh = (this.categoryLabel.length > 0)
      ? this.categoryLabel
      : this.draft.extracted.category_label_zh;
    edited.words = kept;

    await this.client.patchLessonDraft(this.draftId, edited);
    const summary = await this.client.approveLessonDraft(this.draftId);
    this.showToast(`已确认 ${summary.inserted_word_ids.length} 个单词`);
    setTimeout(() => this.back(), 800);
  } catch (e) {
    this.error = this.humanizeError(e);
    if ((e as ParentHttpError).status === 409) {
      setTimeout(() => this.back(), 1000);
    }
  } finally {
    this.busy = false;
  }
}

private async onReject(): Promise<void> {
  this.busy = true;
  this.error = '';
  try {
    await this.client.rejectLessonDraft(this.draftId);
    this.showToast('已丢弃');
    setTimeout(() => this.back(), 600);
  } catch (e) {
    this.error = this.humanizeError(e);
  } finally {
    this.busy = false;
  }
}

private humanizeError(e: object): string {
  const err = e as { name: string; status?: number };
  if (err.name === 'ParentHttpError' && err.status === 409) {
    return '草稿已被处理';
  }
  return '操作失败，请重试';
}
```
