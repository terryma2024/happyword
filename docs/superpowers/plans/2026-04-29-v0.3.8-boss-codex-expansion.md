export const PLAYER_CHARACTER_ASSET: string = 'character/magican.svg';

export function characterAssetForKind(kind: MonsterKind): string {
  if (kind === MonsterKind.Normal) return 'character/slime.svg';
  if (kind === MonsterKind.Spelling || kind === MonsterKind.Review) return 'character/zombie.svg';
  return 'character/dragon.svg';
}

export function assetPathForEntry(entry: MonsterEntry): string {
  if (entry.assetPath !== '') {
    return entry.assetPath;
  }
  return characterAssetForKind(entry.kind);
}

```typescript
private hashSeed(s: string): number {
  let hash: number = 5381;
  for (let i: number = 0; i < s.length; i++) {
    hash = ((hash << 5) - hash) + s.charCodeAt(i);
    hash = hash | 0;
  }
  return Math.abs(hash);
}

private pickBossCatalogIndexForDay(region: AdventureRegion, dayKey: string): number {
  if (region.bossCandidates.length === 0) {
    const slots: MonsterSlot[] = region.monsterPlan.slots;
    return slots[slots.length - 1].catalogIndex;
  }
  const hash: number = this.hashSeed(`${region.id}:${dayKey}`);
  const idx: number = hash % region.bossCandidates.length;
  return region.bossCandidates[idx];
}
```

## 失败回退策略

| 失败位置 | 回退 |
|---|---|
| Phase 1 测试红 | 在 catalog hex 上微调直到 hue / unique 守卫绿；不动其他 phase |
| Phase 2 builder 单测红 | 微调 hash seed / fallback 逻辑直到全绿；BattlePage 改动是 1 行，可独立回滚 |
| Phase 3 codex 测试红 | 微调描述长度 / asset path；不动其他 phase |
| Phase 4 UI 套件回归 | 检查 `MonsterCodexFlow` 的延迟、`returnToHome` 助手；如果只在新用例失败，逐一调试。如果其它套件失败，说明 `assetPathForEntry` 逻辑出现意外侧效应，回到 Phase 2 复查 |
| Phase 4 codelinter 失败 | 跟着提示 `--fix`；如果是 `avoid-overusing-custom-component-check`，说明误用了 `@Component`，改回 `@Builder` |
| 真机崩溃 | 极小概率：可能是巨型 SVG（snow-queen 942KB）触发 ArkUI 上限。回退方案：用 `svgo` 或类似工具压缩到 < 500KB；spec §7 已列为风险 |
